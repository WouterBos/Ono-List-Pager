 10 /**
 11  * @namespace Helper methods for the animation objects.
 12  */
 13 onoPager.tools = (function() {
 14   var HORIZONTAL = 'horizontal';
 15   var VERTICAL = 'vertical';
 16 
 17   return {
 18     /**
 19      * Determines wether a page action is allowed
 20      * @return {Boolean} A page action can be done if return value is true.
 21      * @param {Object} root The root object of a pager (div.onoPager).
 22      * @param {Boolean} arg_lockDuringTransition Determines of a user can make a
 23      *    page before the current transition is finished.
 24      * @param {Object} list The list, most of the times that's a UL.
 25      * @param {Object} listItems The collection of the list items, most of the
 26      *    the times it is a collection of LI's.
 27      */
 28     canPage: function(root,
 29                       lockDuringTransition,
 30                       list,
 31                       listItems) {
 32       if (root.size() > 0 &&
 33           root.hasClass('onoPager_disabled') == false &&
 34           (lockDuringTransition == false ||
 35           lockDuringTransition == true &&
 36           list.is(':animated') == false &&
 37           listItems.is(':animated') == false)) {
 38         return true;
 39       } else {
 40         return false;
 41       }
 42     },
 43 
 44     /**
 45      * Returns either left or top position in pixels relative to the page.
 46      * @param {Object} orientation Either 'horizontal' or 'vertical'.
 47      * @param {Object} selector jQuery selector.
 48      * @return {Number} Left or top offset in pixels.
 49      */
 50     getOffset: function(orientation, selector) {
 51       if (orientation == HORIZONTAL) {
 52         return jQuery(selector).offset().left;
 53       } else if (orientation == VERTICAL) {
 54         return jQuery(selector).offset().top;
 55       }
 56     },
 57 
 58     /**
 59      * Returns either left or top position in pixels relative to its offset
 60      * parent.
 61      * @param {Object} orientation Either 'horizontal' or 'vertical'.
 62      * @param {Object} selector jQuery selector.
 63      * @return {Number} Left or top position in pixels.
 64      */
 65     getPosition: function(orientation, selector) {
 66       if (orientation == HORIZONTAL) {
 67         return jQuery(selector).position().left;
 68       } else if (orientation == VERTICAL) {
 69         return jQuery(selector).position().top;
 70       }
 71     },
 72 
 73     /**
 74      * Returns either the string 'width' or 'height'.
 75      * @param {Object} orientation Either 'horizontal' or 'vertical'.
 76      * @return {String} Either 'width or 'height'.
 77      */
 78     getWidthHeight: function(orientation) {
 79       if (orientation == HORIZONTAL) {
 80         return 'width';
 81       } else if (orientation == VERTICAL) {
 82         return 'height';
 83       }
 84     },
 85 
 86     /**
 87      * Returns either the string 'left' or 'top'.
 88      * @param {Object} orientation Either 'horizontal' or 'vertical'.
 89      * @return {String} Either 'left or 'top'.
 90      */
 91     getTopLeft: function(orientation) {
 92       if (orientation == HORIZONTAL) {
 93         return 'left';
 94       } else if (orientation == VERTICAL) {
 95         return 'top';
 96       }
 97     },
 98 
 99     /**
100      * Returns either the width or height of an element in pixels. Padding,
101      *   border and margin included.
102      * @param {Object} orientation Either 'horizontal' or 'vertical'.
103      * @param {Object} selector jQuery selector.
104      * @return {String} Either width or height in pixels.
105      */
106     getOuterSize: function(orientation, selector, withMargin) {
107       if (orientation == HORIZONTAL) {
108         return jQuery(selector).outerWidth(withMargin);
109       } else if (orientation == VERTICAL) {
110         return jQuery(selector).outerHeight(withMargin);
111       }
112     },
113 
114     /**
115      * Returns either the width or height of an element in pixels. Padding
116      *   included.
117      * @param {Object} orientation Either 'horizontal' or 'vertical'.
118      * @param {Object} selector jQuery selector.
119      * @return {String} Either width or height in pixels.
120      */
121     getInnerSize: function(orientation, selector) {
122       if (orientation == HORIZONTAL) {
123         return jQuery(selector).innerWidth();
124       } else if (orientation == VERTICAL) {
125         return jQuery(selector).innerHeight();
126       }
127     }
128   };
129 })();
