1 /**
  2  * @namespace Helper methods for the animation objects.
  3  */
  4 onoPager.tools = (function() {
  5   var HORIZONTAL = 'horizontal';
  6   var VERTICAL = 'vertical';
  7 
  8   return {
  9     /**
 10      * Returns either left or top position in pixels relative to the page.
 11      * @param {Object} orientation Either 'horizontal' or 'vertical'.
 12      * @param {Object} selector jQuery selector.
 13      * @return {Number} Left or top offset in pixels.
 14      */
 15     getOffset: function(orientation, selector) {
 16       if (orientation == HORIZONTAL) {
 17         return jQuery(selector).offset().left;
 18       } else if (orientation == VERTICAL) {
 19         return jQuery(selector).offset().top;
 20       }
 21     },
 22 
 23     /**
 24      * Returns either left or top position in pixels relative to its offset
 25      * parent.
 26      * @param {Object} orientation Either 'horizontal' or 'vertical'.
 27      * @param {Object} selector jQuery selector.
 28      * @return {Number} Left or top position in pixels.
 29      */
 30     getPosition: function(orientation, selector) {
 31       if (orientation == HORIZONTAL) {
 32         return jQuery(selector).position().left;
 33       } else if (orientation == VERTICAL) {
 34         return jQuery(selector).position().top;
 35       }
 36     },
 37 
 38     /**
 39      * Returns either the string 'width' or 'height'.
 40      * @param {Object} orientation Either 'horizontal' or 'vertical'.
 41      * @return {String} Either 'width or 'height'.
 42      */
 43     getWidthHeight: function(orientation) {
 44       if (orientation == HORIZONTAL) {
 45         return 'width';
 46       } else if (orientation == VERTICAL) {
 47         return 'height';
 48       }
 49     },
 50 
 51     /**
 52      * Returns either the string 'left' or 'top'.
 53      * @param {Object} orientation Either 'horizontal' or 'vertical'.
 54      * @return {String} Either 'left or 'top'.
 55      */
 56     getTopLeft: function(orientation) {
 57       if (orientation == HORIZONTAL) {
 58         return 'left';
 59       } else if (orientation == VERTICAL) {
 60         return 'top';
 61       }
 62     },
 63 
 64     /**
 65      * Returns either the width or height of an element in pixels. Padding,
 66      *   border and margin included.
 67      * @param {Object} orientation Either 'horizontal' or 'vertical'.
 68      * @param {Object} selector jQuery selector.
 69      * @return {String} Either width or height in pixels.
 70      */
 71     getOuterSize: function(orientation, selector) {
 72       if (orientation == HORIZONTAL) {
 73         return jQuery(selector).outerWidth(true);
 74       } else if (orientation == VERTICAL) {
 75         return jQuery(selector).outerHeight(true);
 76       }
 77     },
 78 
 79     /**
 80      * Returns either the width or height of an element in pixels. Padding
 81      *   included.
 82      * @param {Object} orientation Either 'horizontal' or 'vertical'.
 83      * @param {Object} selector jQuery selector.
 84      * @return {String} Either width or height in pixels.
 85      */
 86     getInnerSize: function(orientation, selector) {
 87       if (orientation == HORIZONTAL) {
 88         return jQuery(selector).innerWidth(true);
 89       } else if (orientation == VERTICAL) {
 90         return jQuery(selector).innerHeight(true);
 91       }
 92     }
 93   };
 94 })();
