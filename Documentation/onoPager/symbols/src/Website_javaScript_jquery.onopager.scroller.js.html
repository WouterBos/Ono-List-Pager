 10 /**
 11  * @namespace Manages the scroller control.
 12  *
 13  * @constructor
 14  * @param {Object} arg_pageScroller The scroller, wrapped in a jQuery object.
 15  * @param {Object} arg_listContainer The element that contains the list, wrapped
 16  *    in a jQuery object.
 17  * @param {Object} arg_list The element that contains the list, wrapped
 18  *    in a jQuery object.
 19  * @param {String} arg_orientation Determines on what axis the animation moves.
 20  * Possible values are 'horizontal' and 'vertical' though it's possible to use
 21  * other values as long as the animation object supports that value. Default
 22  * value is 'horizontal'.
 23  */
 24 onoPager.scroller = function(arg_pageScroller,
 25                              arg_listContainer,
 26                              arg_list,
 27                              arg_orientation) {
 28   var pageScroller = arg_pageScroller;
 29   var pageHandle = pageScroller.find('> div.onoPager_scrollerHandle');
 30   var listContainer = arg_listContainer;
 31   var list = arg_list;
 32   var totalScroll;
 33   var updateInterval;
 34   var orientation = arg_orientation;
 35   var noChangeCount = 0;
 36   var previousOffset = 0;
 37   var tools = onoPager.tools;
 38   var listContainerSize = 0;
 39   var listScrollSize = 0;
 40   var interval = 10;
 41   var topLeft = tools.getTopLeft(orientation);
 42 
 43   function _updateHandle() {
 44     var positionList = tools.getOffset(orientation, list);
 45     var positionListContainer = tools.getOffset(orientation, listContainer);
 46     var offset = -(positionList - positionListContainer);
 47     clearIntervalIfNoChange(offset);
 48     var scrollPercentage = Math.round((offset / listScrollSize) * 100);
 49     pageHandle.css(topLeft, ((totalScroll / 100) * scrollPercentage) + 'px');
 50   }
 51 
 52   function clearIntervalIfNoChange(arg_offset) {
 53     if (arg_offset == previousOffset) {
 54       noChangeCount++;
 55       if (noChangeCount > 10) {
 56         clearInterval(updateInterval);
 57       }
 58     } else {
 59       noChangeCount = 0;
 60     }
 61     previousOffset = arg_offset;
 62   }
 63 
 64 
 65 
 66   this.init = function(animation, pageNext, pagePrevious) {
 67     var listSize = tools.getOuterSize(orientation, list, false);
 68     listContainerSize = tools.getInnerSize(orientation, listContainer);
 69     listScrollSize = listSize - listContainerSize;
 70     var sizeKey = tools.getWidthHeight(orientation);
 71     var pageScrollerCss = {};
 72     pageScrollerCss[sizeKey] = listContainerSize + 'px';
 73     if (orientation == 'horizontal') {
 74       pageScrollerCss['height'] = pageHandle.outerHeight();
 75     } else {
 76       pageScrollerCss['width'] = pageHandle.outerWidth();
 77     }
 78     pageScroller.css(pageScrollerCss);
 79 
 80     totalScroll = tools.getInnerSize(orientation, pageScroller) -
 81                     tools.getOuterSize(orientation, pageHandle);
 82     var dragHandler = new onoPager.scroller.dragHandle(pageHandle,
 83                                                        orientation,
 84                                                        totalScroll,
 85                                                        animation,
 86                                                        pageNext,
 87                                                        pagePrevious);
 88     dragHandler.init();
 89   }
 90 
 91   this.updateHandle = function() {
 92     clearInterval(updateInterval);
 93     noChangeCount = 0;
 94     updateInterval = setInterval(_updateHandle, interval, totalScroll);
 95   }
 96 };

103 /**
104  * @namespace Code for making the scroller handle draggable.
105  *
106  * @constructor
107  * @param {Object} arg_handle The handle of the scroller.
108  * @param {String} arg_orientation The orientation of the paer, which is either
109  *    'horizontal' or 'vertical'.
110  * @param {Number} arg_totalScroll The amount of pixels the handle may move
111  *    (which is the size of the scroller minus the size of the handle).
112  * @param {Object} arg_animation The animation object.
113  * @param {Object} arg_pageNext The 'next' button, which has to be disabled when
114  *    you reach the end of the scroller.
115  * @param {Object} arg_pagePrevious The 'previous' button, which has to be
116  *    disabled when you reach the begin of the scroller.
117  */
118 onoPager.scroller.dragHandle = function(arg_handle,
119                                         arg_orientation,
120                                         arg_totalScroll,
121                                         arg_animation,
122                                         arg_pageNext,
123                                         arg_pagePrevious) {
124   var startSize = 0;
125   var offsetSize = 0;
126   var previousSize = 0;
127   var dragElement;
128   var HORIZONTAL = 'horizontal';
129   var VERTICAL = 'vertical';
130   var handle = arg_handle;
131   var orientation = arg_orientation;
132   var totalScroll = arg_totalScroll;
133   var animation = arg_animation;
134   var pageNext = arg_pageNext;
135   var pagePrevious = arg_pagePrevious;
136 
137   function onMouseDown(e) {
138     var target = e.target;
139     if ((e.button == 1 && window.event != null ||
140       e.button == 0)) {
141 
142       if (orientation == HORIZONTAL) {
143         startSize = e.pageX;
144         offsetSize = extractNumber(target.style.left);
145       } else if (orientation == VERTICAL) {
146         startSize = e.pageY;
147         offsetSize = extractNumber(target.style.top);
148       }
149       previousSize = startSize;
150 
151       jQuery(document).bind('mousemove.scroller', onMouseMove);
152       jQuery(document).bind('mouseup.scroller', onMouseUp);
153 
154       // prevent text selection or image drag
155       document.onselectstart = function() {return false;};
156       handle[0].ondragstart = function() {return false;};
157       //document.body.focus();
158       jQuery(handle[0]).closest('div.onoPager_scroller')
159                        .addClass('active');
160       jQuery(handle[0]).closest('div.onoPager_controls')
161                        .addClass('active');
162 
163       return false;
164     }
165   }
166 
167   function onMouseMove(e) {
168     var offset = 0;
169     if (orientation == HORIZONTAL) {
170       offset = setBoundsToOffset(offsetSize + e.pageX - startSize);
171       handle.css('left', offset + 'px');
172     } else if (orientation == VERTICAL) {
173       offset = setBoundsToOffset(offsetSize + e.pageY - startSize);
174       handle.css('top', offset + 'px');
175     }
176     animation.onHandleDrag(Math.round((offset / totalScroll) * 100));
177   }
178 
179   function setBoundsToOffset(arg_offset) {
180     var offset = arg_offset;
181 
182     if (offset < 0) {
183       offset = 0;
184       pagePrevious.addClass('disabled');
185     } else if (offset > totalScroll) {
186       offset = totalScroll;
187       pageNext.addClass('disabled');
188     } else {
189       pageNext.removeClass('disabled');
190       pagePrevious.removeClass('disabled');
191     }
192 
193     return offset;
194   }
195 
196   function onMouseUp(e) {
197     handle[0].ondragstart = null;
198     document.onselectstart = null;
199     jQuery(document).unbind('mousemove.scroller', onMouseMove);
200     jQuery(document).unbind('mouseup.scroller', onMouseUp);
201     jQuery(handle[0]).closest('div.onoPager_scroller')
202                      .removeClass('active');
203     jQuery(handle[0]).closest('div.onoPager_controls')
204                      .removeClass('active');
205 
206   }
207 
208   function extractNumber(value) {
209     var n = parseInt(value);
210     return n == null || isNaN(n) ? 0 : n;
211   }
212 
213 
214 
215   this.init = function() {
216     handle.mousedown(onMouseDown);
217 
218     if (orientation == HORIZONTAL) {
219       handle.css('cursor', 'w-resize');
220     } else if (orientation == VERTICAL) {
221       handle.css('cursor', 'n-resize');
222     }
223   }
224 };
