 15 /**
 16  * @namespace Animation object. Incoming and outgoing slides will always appear
 17  *    next to each other. Even if they are not next to each other in the list,
 18  *    like the first and last item in the list.
 19  *
 20  * @param {Object} newConfig Standard configuration object.
 21  * @param {Object} extraConfig Optional extra configuration object.
 22  * @param {String} extraConfig.color Fill color. Default is '#EB7D2C'.
 23  * @return {Object} instance of an animation object.
 24  */
 25 onoPager.animation.canvas2d_square1 = function(newConfig, arg_extraConfig) {
 26   /**
 27    * New animation object.
 28    * @memberOf onoPager.animation.canvas2d_square1
 29    */
 30   var square1Instance = new onoPager.animation._standard(newConfig,
 31                                                          extraConfig);
 32   if (!square1Instance._config.extraConfig.color) {
 33     square1Instance._config.extraConfig.color = '#EB7D2C';
 34   }
 35   var tools = onoPager.tools;
 36   var drawInterval = null;
 37   var theCanvas;
 38   var interval = 10;
 39   var frames = square1Instance._config.animationSpeed / interval;
 40 
 41   /**
 42    * @see onoPager.animation._standard#init
 43    * @memberOf onoPager.animation.canvas2d_square1
 44    * @this
 45    */
 46   square1Instance.init = function() {
 47     this._config.listItems.css(
 48       {
 49         display: 'none',
 50         position: 'absolute',
 51         top: 0,
 52         left: 0
 53       }
 54     );
 55     jQuery(this._config.listItems[0]).show();
 56   }
 57 
 58   /**
 59    * @see onoPager.animation._standard#page
 60    * @memberOf onoPager.animation.canvas2d_square1
 61    * @this
 62    */
 63   square1Instance.page = function(oldIndex, newIndex, direction) {
 64     var counter = 0;
 65     var context;
 66     var canvasWidth;
 67     var canvasHeight;
 68 
 69     this._config.listItems.hide();
 70     jQuery(this._config.listItems[oldIndex]).show();
 71 
 72     init();
 73 
 74     function init() {
 75       if (typeof(theCanvas) == 'undefined') {
 76         var containerWidth = square1Instance._config.listContainer.outerWidth();
 77         var containerHeight = square1Instance._config.listContainer
 78                                                      .outerHeight();
 79         square1Instance._config.listContainer.append(
 80           '<canvas width="' + containerWidth + '" height="' + containerHeight +
 81           '" style="position: relative;"></canvas>'
 82         );
 83       }
 84       theCanvas = square1Instance._config.listContainer.find('canvas')[0];
 85       context = theCanvas.getContext('2d');
 86       canvasWidth = context.canvas.width;
 87       canvasHeight = context.canvas.height;
 88       if (drawInterval != null) {
 89         resetStage();
 90       }
 91       drawInterval = setInterval(draw, interval);
 92     }
 93 
 94     function resetStage() {
 95       clearInterval(drawInterval);
 96       counter = (frames - 1);
 97       context.clearRect(0, 0, canvasWidth, canvasHeight);
 98     }
 99 
100     function draw() {
101       if (counter <= frames) {
102         context.fillStyle = square1Instance._config.extraConfig.color;
103         var width = parseInt((canvasWidth / frames) * counter);
104         var height = parseInt((canvasHeight / frames) * counter);
105         context.fillRect(
106           parseInt(canvasWidth / 2) - parseInt(width / 2),
107           parseInt(canvasHeight / 2) - parseInt(height / 2),
108           width,
109           height
110         );
111         counter++;
112       } else {
113         clearInterval(drawInterval);
114         counter = 0;
115         jQuery(square1Instance._config.listItems[oldIndex]).hide();
116         jQuery(square1Instance._config.listItems[newIndex]).show();
117         drawInterval = setInterval(clearDraw, interval);
118       }
119     }
120 
121     function clearDraw() {
122       if (counter <= frames) {
123         var width = parseInt((canvasWidth / frames) * counter);
124         var height = parseInt((canvasHeight / frames) * counter);
125         context.clearRect(
126           parseInt(canvasWidth / 2) - parseInt(width / 2),
127           parseInt(canvasHeight / 2) - parseInt(height / 2),
128           width,
129           height
130         );
131         counter++;
132       } else {
133         clearInterval(drawInterval);
134         drawInterval = null;
135       }
136     }
137   }
138

141    * @memberOf onoPager.animation.canvas2d_square1
142    * @this
143    */
144   square1Instance.pagerHover = function(move) {
145     // Not implemented
146   }
147 
148   return square1Instance;
149 };

156 /**
157  * @namespace Animation object. A solid color, swiping clockwise, will obscure
158  *    the viewport. After it recedes, the new slide appears.
159  *
160  * @param {Object} newConfig Standard configuration object.
161  * @param {Object} extraConfig Optional extra configuration object.
162  * @param {String} extraConfig.color Fill color. Default is '#EB7D2C'.
163  * @param {Number} extraConfig.interval The time in milliseconds between screen
164  *    drawings.
165  * @return {Object} instance of an animation object.
166  */
167 onoPager.animation.canvas2d_clock = function(newConfig, extraConfig) {
168   /**
169    * New animation object.
170    * @memberOf onoPager.animation.canvas2d_clock
171    */
172   var clockInstance = new onoPager.animation._standard(newConfig, extraConfig);
173   var extraConfig = clockInstance._config.extraConfig;
174   if (!clockInstance._config.extraConfig.color) {
175     clockInstance._config.extraConfig.color = '#EB7D2C';
176   }
177   if (!clockInstance._config.extraConfig.interval) {
178     clockInstance._config.extraConfig.interval = 10;
179   }
180   var tools = onoPager.tools;
181   var drawClockInterval = null;
182   var theCanvas;
183   var interval = clockInstance._config.extraConfig.interval;
184   var frames = (clockInstance._config.animationSpeed / 2) / interval;
185 
186   /**
187    * @see onoPager.animation._standard#init
188    * @memberOf onoPager.animation.canvas2d_clock
189    * @this
190    */
191   clockInstance.init = function() {
192     this._config.listItems.css(
193       {
194         display: 'none',
195         position: 'absolute',
196         top: 0,
197         left: 0
198       }
199     );
200     jQuery(this._config.listItems[0]).show();
201   }
202 
203   /**
204    * @see onoPager.animation._standard#page
205    * @memberOf onoPager.animation.canvas2d_clock
206    * @this
207    */
208   clockInstance.page = function(oldIndex, newIndex, direction) {
209     var counter = 0;
210     var context;
211     var canvasWidth;
212     var canvasHeight;
213     var degrees;
214 
215     this._config.listItems.hide();
216     jQuery(this._config.listItems[oldIndex]).show();
217 
218     init();
219 
220     function init() {
221       if (typeof(theCanvas) == 'undefined') {
222         var containerWidth = clockInstance._config.listContainer.outerWidth();
223         var containerHeight = clockInstance._config.listContainer.outerHeight();
224         clockInstance._config.listContainer.append(
225           '<canvas width="' + containerWidth + '" height="' + containerHeight +
226           '" style="position: relative;"></canvas>'
227         );
228       }
229       theCanvas = clockInstance._config.listContainer.find('canvas')[0];
230       context = theCanvas.getContext('2d');
231       canvasWidth = context.canvas.width;
232       canvasHeight = context.canvas.height;
233       if (drawClockInterval != null) {
234         resetStage();
235       }
236 
237       degrees = -90;
238       drawClockInterval = setInterval(function() { draw(false) }, interval);
239       jQuery(clockInstance._config.listItems.filter(':visible')).hide();
240       jQuery(clockInstance._config.listItems[oldIndex]).show();
241     }
242 
243     function resetStage() {
244       clearInterval(drawClockInterval);
245       counter = (frames - 1);
246       context.clearRect(0, 0, canvasWidth, canvasHeight);
247     }
248 
249     function draw(oppositeDirection) {
250       degrees += 360 / ((clockInstance._config.animationSpeed / 2) / interval);
251 
252       if (degrees >= 269) {
253         clearInterval(drawClockInterval);
254         if (oppositeDirection == false) {
255           degrees = -90;
256           drawClockInterval = setInterval(function() { draw(true) }, interval);
257           jQuery(clockInstance._config.listItems[oldIndex]).hide();
258           jQuery(clockInstance._config.listItems[newIndex]).show();
259         } else {
260           resetStage();
261         }
262         return;
263       }
264 
265       var centerX = Math.floor(canvasWidth / 2);
266       var centerY = Math.floor(canvasHeight / 2);
267       var radius = Math.floor(canvasWidth);
268       var radian = (Math.PI / 180) * degrees;
269 
270       context.clearRect(0, 0, canvasWidth, canvasHeight);
271       context.beginPath();
272       context.moveTo(centerX, centerY);
273       context.arc(centerX,
274                   centerY,
275                   radius,
276                   (Math.PI / 180) * -90,
277                   radian,
278                   oppositeDirection);
279       context.lineTo(centerX, centerY);
280       context.closePath();
281 
282       context.fillStyle = extraConfig.color;
283       context.fill();
284     }
285 
286     function clearDraw() {
287       if (counter <= frames) {
288         var width = parseInt((canvasWidth / frames) * counter);
289         var height = parseInt((canvasHeight / frames) * counter);
290         context.clearRect(
291           parseInt(canvasWidth / 2) - parseInt(width / 2),
292           parseInt(canvasHeight / 2) - parseInt(height / 2),
293           width,
294           height
295         );
296         counter++;
297       } else {
298         clearInterval(drawClockInterval);
299         drawClockInterval = null;
300       }
301     }
302   }
303

306    * @memberOf onoPager.animation.canvas2d_clock
307    * @this
308    */
309   clockInstance.pagerHover = function(move) {
310     // Not implemented
311   }
312 
313   return clockInstance;
314 };
