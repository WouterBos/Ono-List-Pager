 16  * @namespace Animation object. Incoming and outgoing slides will always appear
 17  *    next to each other. Even if they are not next to each other in the list,
 18  *    like the first and last item in the list.

 23  * @return {Object} instance of an animation object.

 25 onoPager.animation.canvas2d_square1 = function(newConfig, extraConfig) {
 26   /**
 27    * New animation object.
 28    * @memberOf onoPager.animation.canvas2d_square1
 29    */
 30   var square1Instance = new onoPager.animation._standard(newConfig,
 31                                                          extraConfig);
 32   if (!square1Instance._config.extraConfig.color) {
 33     square1Instance._config.extraConfig.color = '#EB7D2C';
 34   }
 35   var tools = onoPager.tools;
 36   var drawInterval = null;
 37   var theCanvas;
 38   var interval = 10;
 39   var frames = square1Instance._config.animationSpeed / interval;
 40 
 41   /**
 42    * @see onoPager.animation._standard#init
 43    * @memberOf onoPager.animation.canvas2d_square1
 44    * @this
 45    */
 46   square1Instance.init = function() {
 47     this._config.listItems.css(
 48       {
 49         display: 'none',
 50         position: 'absolute',
 51         top: 0,
 52         left: 0
 53       }
 54     );
 55     jQuery(this._config.listItems[this._config.activeIndex]).show();
 56   }

 58   /**
 59    * @see onoPager.animation._standard#page
 60    * @memberOf onoPager.animation.canvas2d_square1
 61    * @this
 62    */
 63   square1Instance.page = function(oldIndex, newIndex, direction) {
 64     var counter = 0;
 65     var context;
 66     var canvasWidth;
 67     var canvasHeight;
 68 
 69     this._config.listItems.hide();
 70     jQuery(this._config.listItems[oldIndex]).show();
 71 
 72     init();
 73 
 74     function init() {
 75       if (typeof(theCanvas) == 'undefined') {
 76         square1Instance._config.listContainer.append(
 77           onoPager.tools.createCanvas(square1Instance._config.listContainer
 78                                                              .outerWidth(),
 79                                       square1Instance._config.listContainer
 80                                                              .outerHeight()
 81           )
 82         );
 83       }
 84       theCanvas = square1Instance._config.listContainer.find('canvas')[0];
 85       context = theCanvas.getContext('2d');
 86       canvasWidth = context.canvas.width;
 87       canvasHeight = context.canvas.height;
 88       if (drawInterval != null) {
 89         resetStage();
 90       }
 91       drawInterval = setInterval(draw, interval);
 92     }
 93 
 94     function resetStage() {
 95       clearInterval(drawInterval);
 96       counter = (frames - 1);
 97       context.clearRect(0, 0, canvasWidth, canvasHeight);
 98     }
 99 
100     function draw() {
101       if (counter <= frames) {
102         context.fillStyle = square1Instance._config.extraConfig.color;
103         var width = parseInt((canvasWidth / frames) * counter);
104         var height = parseInt((canvasHeight / frames) * counter);
105         context.fillRect(
106           parseInt(canvasWidth / 2) - parseInt(width / 2),
107           parseInt(canvasHeight / 2) - parseInt(height / 2),
108           width,
109           height
110         );
111         counter++;
112       } else {
113         clearInterval(drawInterval);
114         counter = 0;
115         jQuery(square1Instance._config.listItems[oldIndex]).hide();
116         jQuery(square1Instance._config.listItems[newIndex]).show();
117         drawInterval = setInterval(clearDraw, interval);
118       }
119     }
120 
121     function clearDraw() {
122       if (counter <= frames) {
123         var width = parseInt((canvasWidth / frames) * counter);
124         var height = parseInt((canvasHeight / frames) * counter);
125         context.clearRect(
126           parseInt(canvasWidth / 2) - parseInt(width / 2),
127           parseInt(canvasHeight / 2) - parseInt(height / 2),
128           width,
129           height
130         );
131         counter++;
132       } else {
133         clearInterval(drawInterval);
134         drawInterval = null;
135       }
136     }
137   }

162  * @param {String} extraConfig.color Fill color. Default is '#EB7D2C'.
163  * @param {Number} extraConfig.interval The time in milliseconds between screen
164  *    drawings.
165  * @return {Object} instance of an animation object.

167 onoPager.animation.canvas2d_clock = function(newConfig, extraConfig) {
168   /**
169    * New animation object.
170    * @memberOf onoPager.animation.canvas2d_clock
171    */
172   var clockInstance = new onoPager.animation._standard(newConfig, extraConfig);
173   clockInstance._config.extraConfig = jQuery.extend(
174     {
175       color:'#EB7D2C',
176       interval: 10
177     },
178     clockInstance._config.extraConfig
179   );
180   var tools = onoPager.tools;
181   var drawClockInterval = null;
182   var theCanvas;
183   var interval = clockInstance._config.extraConfig.interval;
184   var frames = (clockInstance._config.animationSpeed / 2) / interval;

187    * @see onoPager.animation._standard#init
188    * @memberOf onoPager.animation.canvas2d_clock

193       {

203   /**
204    * @see onoPager.animation._standard#page
205    * @memberOf onoPager.animation.canvas2d_clock
206    * @this
207    */
208   clockInstance.page = function(oldIndex, newIndex, direction) {
209     var counter = 0;
210     var context;
211     var canvasWidth;
212     var canvasHeight;
213     var degrees;
214 
215     this._config.listItems.hide();
216     jQuery(this._config.listItems[oldIndex]).show();
217 
218     init();
219 
220     function init() {
221       if (typeof(theCanvas) == 'undefined') {
222         clockInstance._config.listContainer.append(
223           onoPager.tools.createCanvas(clockInstance._config.listContainer
224                                                            .outerWidth(),
225                                       clockInstance._config.listContainer
226                                                            .outerHeight()
227           )
228         );
229       }
230       theCanvas = clockInstance._config.listContainer.find('canvas')[0];
231       context = theCanvas.getContext('2d');
232       canvasWidth = context.canvas.width;
233       canvasHeight = context.canvas.height;
234       if (drawClockInterval != null) {
235         resetStage();
236       }
237 
238       degrees = -90;
239       drawClockInterval = setInterval(function() { draw(false) }, interval);
240       jQuery(clockInstance._config.listItems.filter(':visible')).hide();
241       jQuery(clockInstance._config.listItems[oldIndex]).show();
242     }
243 
244     function resetStage() {
245       clearInterval(drawClockInterval);
246       counter = (frames - 1);
247       context.clearRect(0, 0, canvasWidth, canvasHeight);
248     }
249 
250     function draw(oppositeDirection) {
251       degrees += 360 / ((clockInstance._config.animationSpeed / 2) / interval);
252 
253       if (degrees >= 269) {
254         clearInterval(drawClockInterval);
255         if (oppositeDirection == false) {
256           degrees = -90;
257           drawClockInterval = setInterval(function() { draw(true) }, interval);
258           jQuery(clockInstance._config.listItems[oldIndex]).hide();
259           jQuery(clockInstance._config.listItems[newIndex]).show();
260         } else {
261           resetStage();
262         }
263         return;
264       }
265 
266       var centerX = Math.floor(canvasWidth / 2);
267       var centerY = Math.floor(canvasHeight / 2);
268       var radius = Math.floor(canvasWidth);
269       var radian = (Math.PI / 180) * degrees;
270 
271       context.clearRect(0, 0, canvasWidth, canvasHeight);
272       context.beginPath();
273       context.moveTo(centerX, centerY);
274       context.arc(centerX,
275                   centerY,
276                   radius,
277                   (Math.PI / 180) * -90,
278                   radian,
279                   oppositeDirection);
280       context.lineTo(centerX, centerY);
281       context.closePath();
282 
283       context.fillStyle = clockInstance._config.extraConfig.color;
284       context.fill();
285     }
286 
287     function clearDraw() {
288       if (counter <= frames) {
289         var width = parseInt((canvasWidth / frames) * counter);
290         var height = parseInt((canvasHeight / frames) * counter);
291         context.clearRect(
292           parseInt(canvasWidth / 2) - parseInt(width / 2),
293           parseInt(canvasHeight / 2) - parseInt(height / 2),
294           width,
295           height
296         );
297         counter++;
298       } else {
299         clearInterval(drawClockInterval);
300         drawClockInterval = null;
301       }
302     }
303   }

311     // Not implemented

370    * @memberOf onoPager.animation.canvas2d_frost

372    */
373   frostInstance.page = function(oldIndex, newIndex, direction) {
374     var counter = 0;
375     var context;
376     var canvasWidth;
377     var canvasHeight;
378     var diagonalLength;
379 
380     this._config.listItems.hide();
381     jQuery(this._config.listItems[oldIndex]).show();
382 
383     init();
384 
385     function init() {
386       if (typeof(theCanvas) == 'undefined') {
387         frostInstance._config.listContainer.append(
388           onoPager.tools.createCanvas(frostInstance._config.listContainer
389                                                            .outerWidth(),
390                                       frostInstance._config.listContainer
391                                                            .outerHeight()
392           )
393         );
394       }
395       theCanvas = frostInstance._config.listContainer.find('canvas')[0];
396       context = theCanvas.getContext('2d');
397       canvasWidth = context.canvas.width;
398       canvasHeight = context.canvas.height;
399       if (drawInterval != null) {
400         resetStage();
401       }
402       
403       drawInterval = setInterval(
404         function() {
405           draw(false)
406         },
407         interval
408       );
409       jQuery(frostInstance._config.listItems.filter(':visible')).hide();
410       jQuery(frostInstance._config.listItems[oldIndex]).show();
411     }
412 
413     function resetStage() {
414       clearInterval(drawInterval);
415       context.clearRect(0, 0, canvasWidth, canvasHeight);
416     }
417 
418     function draw(oppositeDirection) {
419       if (counter <= frames && counter >= 0) {
420         if (oppositeDirection == true) {
421           counter--;
422         } else {
423           counter++;
424         }
425         var centerX = Math.floor(canvasWidth / 2);
426         var centerY = Math.floor(canvasHeight / 2);
427         diagonalLength = Math.sqrt(centerX*centerX + centerY*centerY);
428         var percentage = (counter / frames) * 100;
429         if (percentage > 100) {
430           percentage = 100;
431         }
432         if (percentage < 0) {
433           percentage = 0;
434         }
435         var stopPosition = 1-(percentage/100);
436         if (stopPosition == 0) {
437           stopPosition = 0.001;
438         }
439         
440         context.clearRect(0, 0, canvasWidth, canvasHeight);
441         var gr = context.createRadialGradient(centerX,centerY,25,centerX,centerY,diagonalLength);
442         opacityEased = (((percentage/100) / 100) * percentage);
443         gr.addColorStop(0, 'rgba(' + color + ', ' + opacityEased + ')');
444         gr.addColorStop(stopPosition, 'rgba(' + color + ', ' + (percentage/100) + ')');
445         gr.addColorStop(1, 'rgba(' + color + ', 1)');
446   
447         context.fillStyle = gr;
448         context.fillRect(0, 0, canvasWidth, canvasHeight);
449       } else {
450         clearInterval(drawInterval);
451         if (oppositeDirection == false) {
452           jQuery(frostInstance._config.listItems[oldIndex]).hide();
453           jQuery(frostInstance._config.listItems[newIndex]).show();
454           counter = frames;
455           drawInterval = setInterval(
456             function() {
457               draw(true)
458             },
459             interval
460           );
461         } else {
462           context.clearRect(0, 0, canvasWidth, canvasHeight);
463         }
464       }
465     }
466   }

502   var extraConfig = jQuery.extend(
503     {
504       color:'235,125,44',
505       interval: 10,
506       freezeUntilImagesPreloaded: false,
507       bgColor: '#ffffff'
508     },
509     imageGridInstance._config.extraConfig
510   );
511   var timeouts = new Array();
512   var config = imageGridInstance._config;
513   var tools = onoPager.tools;
514   var drawInterval = null;
515   var theCanvas;
516   var images = new Array();
517   var frames = (config.animationSpeed / 2) / extraConfig.interval;

519   /**
520    * @see onoPager.animation._standard#init
521    * @memberOf onoPager.animation.canvas2d_imageGrid
522    * @this
523    */
524   imageGridInstance.init = function() {
525     if (typeof(theCanvas) == 'undefined') {
526       config.listContainer.append(
527         onoPager.tools.createCanvas(config.listContainer.outerWidth(),
528                                     config.listContainer.outerHeight()
529         )
530       );
531     }
532     theCanvas = config.listContainer.find('canvas')[0];
533     theCanvas.style.position = 'static';
534     context = theCanvas.getContext('2d');
535     canvasWidth = context.canvas.width;
536     canvasHeight = context.canvas.height;
537 
538     this._config.listItems.css(
539       {
540         display: 'none',
541         position: 'absolute',
542         top: 0,
543         left: 0
544       }
545     );
546     jQuery(this._config.listItems[this._config.activeIndex]).show();
547     
548     preloadImages();
549     
550     // Preload all images that are needed for the transistions
551     function preloadImages() {
552       var imageTotal = 0;
553       var imageCount = 0;
554       
555       if (extraConfig.freezeUntilImagesPreloaded == true) {
556         config.listContainer.parent().addClass('onoPager_disabled');
557       }
558       imageTotal = config.listItems.filter('[data-onopagerbackgroundimage]').size();
559       config.listItems.each(function() {
560         if (jQuery(this).attr('data-onopagerbackgroundimage') == "") {
561           images.push(null);
562         } else {
563           var img = new Image();
564           if (extraConfig.freezeUntilImagesPreloaded == true) {
565             img.addEventListener('load', imgCounter , false);
566           }
567           img.src = jQuery(this).attr('data-onopagerbackgroundimage');
568           images.push(img);
569         }
570       });
571       
572       // Unlock pager if all images are either loaded or failed loading
573       function imgCounter(e) {
574         imageCount++;
575         if (imageCount == imageTotal) {
576           config.listContainer.parent().removeClass('onoPager_disabled');
577           
578           context.drawImage(images[config.activeIndex], 0, 0);
579         }
580       }
581     }
582   }

584   /**
585    * @see onoPager.animation._standard#page
586    * @memberOf onoPager.animation.canvas2d_imageGrid
587    * @this
588    */
589   imageGridInstance.page = function(oldIndex, newIndex, direction) {
590     init();
591 
592     // Runs animation
593     function init() {
594       if (drawInterval != null) {
595         resetStage();
596       }
597       var contentFadeSpeed = config.animationSpeed/6
598       var contentAniSpeedStart;
599       var canvasAniSpeed = config.animationSpeed - (contentFadeSpeed * 2);
600 
601       config.listItems.stop(true, true)
602                       .hide();
603       if (timeouts.length == 0) {
604         // There's no transition still in progress
605         jQuery(config.listItems[oldIndex]).show()
606                                           .fadeOut(contentFadeSpeed);
607         contentAniSpeedStart = contentFadeSpeed;
608       } else {
609         contentAniSpeedStart = 0;
610       }
611       
612       // Set variables
613       var maxTransitionDimensions = getMaxTransitionDimensions();
614       clearTimeouts(timeouts);
615       timeouts = new Array();
616       var min = Math.min(maxTransitionDimensions.width,
617                          maxTransitionDimensions.height)
618       min = Math.ceil(min / extraConfig.gridSize);
619       var max = Math.max(maxTransitionDimensions.width,
620                          maxTransitionDimensions.height)
621       max = Math.ceil(max / extraConfig.gridSize);
622       var loopTotal = (min + ((max - min) / 2)) * 2;
623       console.log(min, max, loopTotal)
624       var interval = canvasAniSpeed / loopTotal;
625       
626       // draws the image of the current list item
627       context.drawImage(images[oldIndex], 0, 0);
628       
629       // Animates the image of the new list item
630       timeouts.push(setTimeout(
631         function() {
632           for (var i = 0; i < loopTotal; i++) {
633             // Draws the diagonal top-left / bottom right
634             (function(i) {
635               timeouts.push(setTimeout(
636                 function() {
637                   context.drawImage(
638                     images[newIndex],
639                     i * extraConfig.gridSize,
640                     i * extraConfig.gridSize,
641                     extraConfig.gridSize,
642                     extraConfig.gridSize,
643                     i * extraConfig.gridSize,
644                     i * extraConfig.gridSize,
645                     extraConfig.gridSize,
646                     extraConfig.gridSize
647                   );
648                 },
649                 (i * interval)
650               ));
651     
652               var total = i
653               for (var ii = 0; ii <= total; ii++) {
654                 // After each block has been drawn of the diagonal
655                 // top-left / bottom right, a complete diagonal is drawn from
656                 // bottom-left to top-right.
657                 (function(ii) {
658                   timeouts.push(setTimeout(
659                     function() {
660                       var x = (total - ii) * extraConfig.gridSize;
661                       var y = ii * extraConfig.gridSize;
662                       context.drawImage(
663                         images[newIndex],
664                         x,
665                         y,
666                         extraConfig.gridSize,
667                         extraConfig.gridSize,
668                         x,
669                         y,
670                         extraConfig.gridSize,
671                         extraConfig.gridSize
672                       );
673                     },
674                     ((i * interval) / 2)
675                   ));
676                 })(ii);
677               }
678             })(i);
679           }
680         },
681         contentAniSpeedStart
682       ));
683       
684       timeouts.push(setTimeout(
685         function() {
686           jQuery(config.listItems[newIndex]).fadeIn(contentFadeSpeed);
687           timeouts = new Array();
688         },
689         canvasAniSpeed
690       ));
691 
692       function clearTimeouts(timeouts) {
693         for (var i = 0; i < timeouts.length; i++){
694           clearTimeout(timeouts[i]);
695         };
696       }
697       
698       function getMaxTransitionDimensions() {
699         var dimensions = {};
700         var oldListItem = jQuery(config.listItems[oldIndex]);
701         var newListItem = jQuery(config.listItems[newIndex]);
702         dimensions.width = Math.max(newListItem.outerWidth(true),
703                                     config.listContainer.outerWidth(true));
704         dimensions.height = Math.max(newListItem.outerHeight(true),
705                                      config.listContainer.outerHeight(true));
706         console.log(dimensions)
707         return dimensions;
708       }
709     }
710 
711     function resetStage() {
712       clearInterval(drawInterval);
713       context.clearRect(0, 0, canvasWidth, canvasHeight);
714     }
715   }
