 37   /**
 38    * Creates a widget that enables the user to navigate through a list of
 39    * elements (like an <code><ul></code>). It provides different types of
 40    * paging navigation. These are called animation objects and reside in the
 41    * <code>onoPager.animation</code> namespace.<br />
 42    * The onoPager library consist of many types of animations like "slides",
 43    * "linear" and "linearScroller". The onoPager is built in such a way that
 44    * it's easy to add your own animation objects without having to rewrite the
 45    * basic paging functionality.<br />
 46    * Check out some <a href="http://www.thebrightlines.com/onopager/website/">
 47    * configuration examples</a>.
 48    *
 49    * @param {Object} arg_config Configuration object.
 50    *
 51    * @param {Object} animationConfig Optional extra configuration object for
 52    *    the animation object.
 53    *
 54    * @param {String} arg_config.cssClass Ads CSS class to the root of the pager.
 55    *
 56    * @param {Boolean} arg_config.pagePerItem If true, each page will be as long
 57    *    as one list item. If false, each page will be as long as the list
 58    *    container, which may span multiple list items.
 59    *
 60    * @param {Boolean} arg_config.lockDuringTransition Disable paging controls
 61    *    during a transition when true. Default is false.
 62    *
 63    * @param {Boolean} arg_config.doesLoop If true, the pager scrolls back
 64    *    to the first item after the last item.
 65    *
 66    * @param {String} arg_config.listContainer.width Width of list
 67    *    container, like '200px'.
 68    *
 69    * @param {String} arg_config.listContainer.height Height of list
 70    *    container, like '200px'.
 71    *
 72    * @param {String} arg_config.listContainer.maxHeight Maximum height of list
 73    *    container, like '200px'. The maximum height will also be applied to all
 74    *    list Items.
 75    *
 76    * @param {Boolean} arg_config.listContainer.adjustHeightToListItem.active If
 77    *    you also set pagePerItem to true, the height of the list container will
 78    *    adjust to the height of the visible list item. Default value is true.
 79    *
 80    * @param {Boolean} arg_config.listContainer.adjustHeightToListItem.animate If
 81    *    true, the container will animatie to its new height. Default value is
 82    *    true.
 83    *
 84    * @param {String} arg_config.ListItems.width Width of list items, like
 85    *    '200px'.
 86    *
 87    * @param {String} arg_config.ListItems.height Height of list items,
 88    *    like '200px'.
 89    *
 90    * @param {Number} arg_config.activeIndex Sets initial visible page. By
 91    *    default the pager starts at index 0.
 92    *
 93    * @param {Boolean} arg_config.autoPage.active Activates auto pager.
 94    *
 95    * @param {Number} arg_config.autoPage.interval The interval between
 96    *    autopaging. Time value is set in milliseconds.
 97    *
 98    * @param {Boolean} arg_config.autoPage.active Activates auto pager.
 99    *
100    * @param {String} arg_config.autoPage.autoPageAnimationType The type
101    *    of animation that will indicate the time the time between transitions.
102    *
103    * @param {Object} arg_config.autoPage.extraConfig A configuration object
104    *    for the autopage indicator.
105    *
106    * @param {Boolean} arg_config.autoPage.pauseOnHover Stops the pager when
107    *    the mouse cursor hovers above the pager.
108    *
109    * @param {String} arg_config.labels.next text for the 'next'-button.
110    *
111    * @param {String} arg_config.labels.previous Text for the
112    *    'previous'-button.
113    *
114    * @param {Boolean} arg_config.status.active Activates the status box.
115    *
116    * @param {String} arg_config.status.prependText Text that appears before
117    *    the page index number.
118    *
119    * @param {String} arg_config.status.seperationText Text that appears
120    *    between the page index number and the total pages number.
121    *
122    * @param {String} arg_config.status.appendText Text that appears after
123    *    the total pages number.
124    *
125    * @param {Boolean} arg_config.scroller.active Activates a Javascript
126    *    scrollbar. Default is true.
127    *
128    * @param {Number} arg_config.pixelMove The amount of pixels the pager
129    *    scrolls after each frame.
130    *
131    * @param {Boolean} arg_config.pageByNumber.active Activates the bar with
132    *    all pages, defined by number. Default is true.
133    *
134    * @param {Boolean} arg_config.pageByNumber.enableClick Disables paging
135    *    behaviour onclick. The makes the 'Page by number'-box essentially a
136    *    status box rather than a navigation control. Default is true.
137    *
138    * @param {Array} arg_config.pageByNumber.labels Replaces the default
139    *    content of the page-by-number links with the string values in the array.
140    *
141    * @param {Number} arg_config.pageByNumber.hideThreshold Defines how many
142    *    links may appear in the pageByNumber box. If you set the value to 2,
143    *    you'll see 2 links on the left and 2 on the right of the active item. A
144    *    negative value cancels this functionality. Default value is -1.
145    *
146    * @param {Array} arg_config.pageByNumber.links Overrides the default click
147    *    behavior of the page-by-number links. If you supply an array of URL's,
148    *    they will behave like standard links that point to an URL.
149    *
150    * @param {Boolean} arg_config.pageByNumber.enableHover Initiates a page
151    *    when the mouse is hovered over a page by number item.
152    *
153    * @param {Boolean} arg_config.pageByArrowKey.active Enables paging by
154    *    using the keyboard arrow keys. Default is false.
155    *
156    * @param {Boolean} arg_config.pageByArrowKey.preventDefault Disables or
157    *    activates the default behaviour of the arrow key. If set to true, the
158    *    user won't be able to scroll the page or a textarea with the arrow keys.
159    *    Default is false for that reason.
160    *
161    * @param {Boolean} arg_config.swipeTriggersPage Activates page
162    *    navigation by swiping on the screen. Default is true.
163    *
164    * @param {String} arg_config.swipePlatforms Determines on what platforms
165    *    the user is able to page by swiping. 'touch' activates swiping only on
166    *    touch devices. 'all' will activates swiping on touch devices and
167    *    desktop. Swiping on the desktop is done with mouse gestures. Default
168    *    value is 'touch'.
169    *
170    * @param {String} arg_config.animationType Determines which animation
171    *    object will be used. The following animation types are available by
172    *    default: 'linear', 'linearScroller' and 'slides'. Custom animation
173    *    objects can be created and used after the plugin is loaded. Default
174    *    value is 'linear'.
175    *
176    * @param {String} arg_config.animationEasing Determines the easing type
177    *    to be used by the animation object. Default value is 'linear'. If
178    *    'easeOutCubic' is available, it will use that as default value.
179    *
180    * @param {String} arg_config.orientation Determines on what axis the
181    *    animation moves. Possible values are 'horizontal' and 'vertical' though
182    *    it's possible to use other values as long as the animation object
183    *    supports that value. Default value is 'horizontal'.
184    *
185    * @param {Number} arg_config.animationSpeed Determines the speed at
186    *    which the animations take place.
187    *
188    * @return {jQuery} chainable jQuery class.
189    * @memberOf jQuery.fn
190    *
191    * @example
192    * // Simple example:
193    * // More examples on: http://www.thebrightlines.com/onopager/website/
194    * jQuery('#list1').onoPager({
195    *   cssClass: 'onoPager_greyscale',
196    *   pagePerItem: true,
197    *   doesLoop: false,
198    *   listContainer: {
199    *     width: '280px',
200    *     height: ''
201    *   },
202    *   listItems: {
203    *     width: '260px',
204    *     height: '80px'
205    *   },
206    *   status: {
207    *     active: false
208    *   },
209    *   pageByArrowKey: {
210    *     active: true,
211    *     preventDefault: false
212    *   },
213    *   pageByNumber: {
214    *     hideThreshold: 1
215    *   },
216    *   swipeTriggersPage: true,
217    *   animationType: 'linear',
218    *   animationEasing: 'easeOutCubic',
219    *   orientation: 'horizontal',
220    *   animationSpeed: 1000
221    * });
222    */

416     // Creates pager object (onoPager.pager)
417     function setPager() {
418       var pageTotal = animation.getPagesLength();
419       if (typeof(pageTotal) != 'number') {
420         throw new Error(
421           'getPagesLength() does not return a number but ' +
422           typeof(pageTotal)
423         );
424       }
425       if (pageTotal < 0) {
426         throw new Error('getPagesLength() must not return a negative number');
427       }
428       if (pageTotal > 1) {
429         pager = new onoPager.pager(
430           config.activeIndex,
431           pageTotal,
432           config.doesLoop,
433           {
434             next: pageNext,
435             previous: pagePrevious,
436             gotoLinks: pageByNumber,
437             status: pageStatus
438           },
439           config.status,
440           config.pageByNumber.hideThreshold
441         );
442       } else {
443         pageNext.hide();
444         pagePrevious.hide();
445         pageByNumber.hide();
446         pageScroller.hide();
447         autoPageContainer.hide();
448       }
449       animation.extendConfig({pager: pager});
450       animation._onPagerCreated();
451       if (config.autoPage.active == true && pageTotal > 1) {
452         pager.initAutopager(config.autoPage,
453                             config.animationSpeed,
454                             animation,
455                             config.orientation,
456                             listContainer,
457                             list,
458                             autoPageContainer,
459                             config.lockDuringTransition);
460       }
461     }

463     // Creates a list of quicklinks to navigate to a specific list item
464     function setPageByNumber() {
465       var pageTotal = animation.getPagesLength();
466       var html = '';
467       var label;
468       var href;
469       var hrefVoid = 'javascript:void(0)';
470 
471       for (var i = 0; i < pageTotal; i++) {
472         // Set label text
473         if (config.pageByNumber.labels &&
474             config.pageByNumber.labels.length > i) {
475           label = config.pageByNumber.labels[i];
476         } else {
477           label = i + 1;
478         }
479 
480         // Set label link
481         if (config.pageByNumber.links &&
482             config.pageByNumber.links.length > i &&
483             config.pageByNumber.links[i] != '') {
484           href = config.pageByNumber.links[i];
485         } else {
486           href = hrefVoid;
487         }
488 
489         html += '<a href="' + href + '"><span>' + label + '</span></a>';
490       }
491 
492       pageByNumber.html(html);

494       setPageEvent();

496       function setPageEvent() {
497         var eventTypes = '';
498         // TODO: change 'enableClick' into something like: interactive
499         if (config.pageByNumber.enableClick == true) {
500           pageByNumber.find('a').each(function(index) {
501             $(this).click(function(event) {
502               var isActive = $(this).hasClass('onoPager_active');
503               if (isActive == false && $(this).attr('href') == hrefVoid) {
504                 page(index);
505               }
506             });
507           });
508         }
509         if (config.pageByNumber.enableHover == true) {
510           pageByNumber.find('a').each(function(index) {
511             $(this).mouseenter(function(event) {
512               if ($(this).hasClass('onoPager_active') == false) {
513                 page(index);
514               }
515             });
516           });
517         }
518 
519         if (config.pageByNumber.enableClick == false &&
520             config.pageByNumber.enableHover == false) {
521           pageByNumber.find('a').addClass('onoPager_readonly');
522         }
523       }
524     }

526     // Attach events to the navigation controls. Without it, they won't respond.
527     function setControlEvents() {
528       // Page with scroller
529       if (config.scroller.active == true) {
530 
531         var scroller = new onoPager.scroller(pageScroller,
532                                              listContainer,
533                                              list,
534                                              config.orientation);
535         scroller.init(animation, pageNext, pagePrevious);
536         animation.extendConfig({scroller: scroller});
537       }
538 
539       // Page by clicking on paging button
540       pageNext.click(function() {
541         page((pager.getIndex() + 1), 1);
542       });
543       pagePrevious.click(function() {
544         page((pager.getIndex() - 1), -1);
545       });
546       if (config.listItems.triggersPagingOnClick == true) {
547         listContainer.find(' > li, .onoPager_listItem').click(function() {
548           var currentIndex = pager.getIndex();
549           var newIndex = parseInt(jQuery(this)
550                                   .attr('data-onopager-list-index'));
551           var direction = 1;
552           if (jQuery(this).attr('data-onopager-list-direction')) {
553             direction = jQuery(this).attr('data-onopager-list-direction');
554           } else if (newIndex < currentIndex) {
555             direction = -1;
556           }
557           page(newIndex, direction);
558         });
559       }
560 
561       // Set hover events on paging buttons
562       pageNext.mouseenter(function() {
563         pagerHover(config.scroller.pixelMove);
564       });
565       pageNext.mouseleave(function() {
566         pagerHover(0);
567       });
568       pagePrevious.mouseenter(function() {
569         pagerHover(-config.scroller.pixelMove);
570       });
571       pagePrevious.mouseleave(function() {
572         pagerHover(0);
573       });
574 
575       // Page with arrow keys on keyboard
576       if (config.pageByArrowKey.active == true) {
577         $(document).keydown(function(event) {
578           var key = event.which;
579           var UP = 38;
580           var DOWN = 40;
581           var LEFT = 37;
582           var RIGHT = 39;
583           // Page back
584           if ((config.orientation == VERTICAL && key == UP) ||
585             (config.orientation == HORIZONTAL && key == LEFT)) {
586             if (preventPage() == false) {
587               page((pager.getIndex() - 1), -1);
588             }
589             if (preventDefaultKeyEvent() == true) {
590               event.preventDefault();
591             }
592           }
593           // Page forward
594           if ((config.orientation == VERTICAL && key == DOWN) ||
595             (config.orientation == HORIZONTAL && key == RIGHT)) {
596             if (preventPage() == false) {
597               page((pager.getIndex() + 1), 1);
598             }
599             if (preventDefaultKeyEvent() == true) {
600               event.preventDefault();
601             }
602           }
603         });
604       }
605 
606       function preventDefaultKeyEvent() {
607         if (config.pageByArrowKey.preventDefault == true) {
608           return true;
609         }
610         return false;
611       }
612 
613       // No paging when the cursor is in a form field,
614       function preventPage() {
615         if (jQuery('textarea:focus, select:focus, input:focus').size() > 0) {
616           return true;
617         }
618         return false;
619       }
620 
621       // Page with swiping
622       if (config.swipeTriggersPage == true) {
623         if (config.orientation == HORIZONTAL) {
624           listContainer.onoPagerSwipe(
625             {
626               dragRight: function() {
627                 page((pager.getIndex() - 1), -1);
628               },
629               dragLeft: function() {
630                 page((pager.getIndex() + 1), 1);
631               },
632               platform: config.swipePlatforms,
633               pageDirection: config.orientation
634             }
635           );
636         } else if (config.orientation == VERTICAL) {
637           listContainer.onoPagerSwipe(
638             {
639               dragDown: function() {
640                 page((pager.getIndex() - 1), -1);
641               },
642               dragUp: function() {
643                 page((pager.getIndex() + 1), 1);
644               },
645               platform: config.swipePlatforms,
646               pageDirection: config.orientation
647             }
648           );
649         }
650       }
651     }

653     // Keeps track of the index of the active list item and passes the page
654     // call to the pager method of the animation instance.
655     function page(arg_newIndex, arg_direction) {
656       var canPage = onoPager.tools.canPage(root,
657                                            config.lockDuringTransition,
658                                            list,
659                                            listItems);
660       if (canPage) {
661         if (config.autoPage.active == true) {
662           pager.resetAutopager();
663         }
664         var oldIndex = pager.getIndex();
665         var newIndex = pager.setIndex(arg_newIndex);
666         animation._page(oldIndex, newIndex, arg_direction);
667       }
668     }

670     // Detect browser so you can write CSS fallbacks for MSIE. Yeah, I know
671     // browser detection feels dirty, but I know you'll thank me in the end :)
672     function addBrowserClass() {
673       var uaClass = '';
674       if (navigator.appName == 'Microsoft Internet Explorer') {
675         uaClass = 'msie' + getInternetExplorerVersion();
