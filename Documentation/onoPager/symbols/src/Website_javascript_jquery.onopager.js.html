 38   /**
 39    * Creates a widget that enables the user to navigate through a list of
 40    * elements (like an <code><ul></code>). It provides different types of
 41    * paging navigation. These are called animation objects and reside in the
 42    * <code>onoPager.animation</code> namespace. The onoPager library consist
 43    * now of 3 types of animations: "slides", "linear" and "linearScroller".
 44    * The onoPager is built in such a way that it's easy to add your own
 45    * animation objects without having to rewrite the basic paging functionality.
 46    *
 47    * @param {Object} arg_config Configuration object.
 48    * @param {Object} animationConfig Optional extra configuration object for
 49    *    the animation object.
 50    * @param {Boolean} animationConfig.pagePerItem Page per item.
 51    * @param {Boolean} animationConfig.pageLoop If true, the pager scrolls back
 52    *    to the first item after the last item.
 53    * @param {String} animationConfig.ListContainer.width Width of list
 54    *    container, like '200px'.
 55    * @param {String} animationConfig.ListContainer.height Height of list
 56    *    container, like '200px'.
 57    * @param {String} animationConfig.ListItems.width Width of list items, like
 58    *    '200px'.
 59    * @param {String} animationConfig.ListItems.height Height of list items,
 60    *    like '200px'.
 61    * @param {Number} animationConfig.activeIndex Sets initial visible page. By
 62    *    default the pager starts at index 0.
 63    * @param {Boolean} animationConfig.autoPage.active Activates auto pager.
 64    * @param {Number} animationConfig.autoPage.interval The interval between
 65    *    autopaging. Time value is set in milliseconds.
 66    * @param {String} animationConfig.labels.next text for the 'next'-button.
 67    * @param {String} animationConfig.labels.previous Text for the
 68    *    'previous'-button.
 69    * @param {Boolean} animationConfig.status.active Activates the status box.
 70    * @param {String} animationConfig.status.prependText Text that appears before
 71    *    the page index number.
 72    * @param {String} animationConfig.status.seperationText Text that appears
 73    *    between the page index number and the total pages number.
 74    * @param {String} animationConfig.status.appendText Text that appears after
 75    *    the total pages number.
 76    * @param {Boolean} animationConfig.scroller.active Activates a Javascript
 77    *    scrollbar. Default is true.
 78    * @param {Boolean} animationConfig.scroller.active Activates a Javascript
 79    *    scrollbar. Default is true.
 80    * @param {Boolean} animationConfig.pixelMove The amount of pixels the pager
 81    *    scrolls after each frame.
 82    * @param {Boolean} animationConfig.pageByNumber.active Activates the bar with
 83    *    all pages, defined by number. Default is true.
 84    * @param {Boolean} animationConfig.swipeTriggersPage Activates page
 85    *    navigation by swiping on the screen. Default is false.
 86    * @param {String} animationConfig.swipePlatforms Determines on what platforms
 87    *    the user is able to page by swiping. 'touch' activates swiping only on
 88    *    touch devices. 'all' will activates swiping on touch devices and
 89    *    desktop. Swiping on the desktop is done with mouse gestures. Default
 90    *    value is 'touch'.
 91    * @param {String} animationConfig.animationType Determines which animation
 92    *    object will be used. The following animation types are available by
 93    *    default: 'linear', 'linearScroller' and 'slides'. Custom animation
 94    *    objects can be created and used after the plugin is loaded.
 95    * @param {String} animationConfig.animationEasing Determines the easing type
 96    *    to be used by the animation object. Default value is 'linear'.
 97    * @param {String} animationConfig.orientation Determines on what axis the
 98    *    animation moves. Possible values are 'horizontal' and 'vertical' though
 99    *    it's possible to use other values as long as the animation object
100    *    supports that value. Default value is 'horizontal'.
101    * @param {Number} animationConfig.animationSpeed Determines the speed at
102    *    which the animations take place.
103    * @return {jQuery} chainable jQuery class.
104    * @memberOf jQuery.fn
105    * @example
106    * // Simple example:
107    * jQuery('#list1').onoPager({
108    *   pagePerItem: false,
109    *   listContainer: {
110    *     width: '700px',
111    *     height: '100px'
112    *   },
113    *   listItems: {
114    *     width: '300px'
115    *   },
116    *   animationType: 'linear',
117    *   animationSpeed: 1000
118    * });
119    *
120    * // Advanced example:
121    * jQuery('#list1').onoPager({
122    *    pagePerItem: true,
123    *    pageLoop: true,
124    *    listContainer: {
125    *      width: '300px',
126    *      height: '100px'
127    *    },
128    *    listItems: {
129    *      width: '300px',
130    *      height: '100px'
131    *    },
132    *    activeIndex: 2,
133    *    autoPage: {
134    *      active: true,
135    *      interval: 2000,
136    *    },
137    *    labels: {
138    *      next: 'next',
139    *      previous: 'previous'
140    *    },
141    *    status: {
142    *      active: true,
143    *      prependText: 'page ',
144    *      seperationText: ' of ',
145    *      appendText: ' pages'
146    *    },
147    *    scroller: {
148    *      active: false,
149    *      pixelMove: 2
150    *    },
151    *    pageByNumber: {
152    *      active: true
153    *    },
154    *    keyTriggersPage: false,
155    *    swipeTriggersPage: false,
156    *    swipePlatforms: 'touch',
157    *    animationType: 'linear',
158    *    animationEasing: 'linear',
159    *    orientation: 'horizontal',
160    *    animationSpeed: 1000
161    * });
162    */

183       },

242     // Create pager controls like 'next' and 'previous'
243     function createControls() {
244       var newHTML = '';
245       newHTML += '<a' + EMPTY_HREF +
246         ' class="' + ONOPAGER + '_previous ' + ONOPAGER + '_step" title="' +
247         config.labels.previous + '">' + config.labels.previous + '</a>';
248       if (config.pageByNumber.active == true) {
249         newHTML += '<div class="' + ONOPAGER + '_pageByNumber"/>';
250       }
251       newHTML += '<a' + EMPTY_HREF + ' class="' + ONOPAGER + '_next ' +
252         ONOPAGER + '_step" ' + 'title="' + config.labels.next + '">' +
253         config.labels.next + '</a>';
254       if (config.status.active == true) {
255         newHTML += '<div class="' + ONOPAGER + '_status" />';
256       }

258         newHTML += '<div class="' + ONOPAGER + '_scroller"><div class="' +

260       }

262         '<div class="' + ONOPAGER + '_controls">' + newHTML + '</div>'

266       pagePrevious = root.find('> * > a.' + ONOPAGER + '_previous');
267       pageByNumber = root.find('> * > div.' + ONOPAGER + '_pageByNumber');
268       pageStatus = root.find('> * > div.' + ONOPAGER + '_status');
269       pageScroller = root.find('> * > div.' + ONOPAGER + '_scroller');

294     function setPager() {
295       var pageTotal = animation.getPagesLength();
296       if (typeof(pageTotal) != 'number') {
297         throw new Error(
298           'getPagesLength() does not return a number but ' +
299           typeof(pageTotal)
300         );
301       }
302       if (pageTotal < 0) {
303         throw new Error('getPagesLength() must not return a negative number');
304       }
305       if (pageTotal > 1) {
306         pager = new onoPager.pager(
307           config.activeIndex,
308           pageTotal,
309           config.pageLoop,
310           {
311             next: pageNext,
312             previous: pagePrevious,
313             gotoLinks: pageByNumber,
314             status: pageStatus
315           },
316           config.status
317         );
318         if (config.autoPage.active == true) {
319           pager.initAutopager(config.autoPage,
320                               animation,
321                               config.orientation,
322                               listContainer,
323                               list);
324         }
325       } else {
326         pageNext.hide();
327         pagePrevious.hide();
328       }
329       animation.extendConfig({pager: pager});
330       animation._onPagerCreated();
331     }

333     function setPageByNumber() {
334       var pageTotal = animation.getPagesLength();
335       var html = '';
336

339       }

342       pageByNumber.find('a').each(function(index) {
343         $(this).click(function() {
344           page(index);
345         });
346       });

349     function setControlEvents() {
350       // Page with scroller
351       if (config.scroller.active == true) {
352 
353         var scroller = new onoPager.scroller(pageScroller,
354                                              listContainer,
355                                              list,
356                                              config.orientation);
357         scroller.init(animation, pageNext, pagePrevious);
358         animation.extendConfig({scroller: scroller});
359       }
360 
361       // Page by clicking on paging button
362       pageNext.click(function() {
363         page(pager.getIndex() + 1);
364       });
365       pagePrevious.click(function() {
366         page(pager.getIndex() - 1);
367       });
368 
369       // Set hover events on paging buttons
370       pageNext.mouseenter(function() {
371         pagerHover(config.scroller.pixelMove);
372       });
373       pageNext.mouseleave(function() {
374         pagerHover(0);
375       });
376       pagePrevious.mouseenter(function() {
377         pagerHover(-config.scroller.pixelMove);
378       });
379       pagePrevious.mouseleave(function() {
380         pagerHover(0);
381       });
382 
383       // Page with arrow keys on keyboard
384       if (config.keyTriggersPage == true) {
385         $(document).keydown(function(event) {
386           var key = event.which;
387           var UP = 38;
388           var DOWN = 40;
389           var LEFT = 37;
390           var RIGHT = 39;
391           // Page back
392           if ((config.orientation == VERTICAL && key == UP) ||
393             (config.orientation == HORIZONTAL && key == LEFT)) {
394             page(pager.getIndex() - 1);
395             event.preventDefault();
396             event.stopPropagation();
397           }
398           // Page forward
399           if ((config.orientation == VERTICAL && key == DOWN) ||
400             (config.orientation == HORIZONTAL && key == RIGHT)) {
401             page(pager.getIndex() + 1);
402             event.preventDefault();
403             event.stopPropagation();
404           }
405         });
406       }
407 
408       // Page with swiping
409       if (config.swipeTriggersPage == true) {
410         if (config.orientation == HORIZONTAL) {
411           listContainer.onoPagerSwipe(
412             {
413               dragRight: function() {
414                 page(pager.getIndex() - 1);
415               },
416               dragLeft: function() {
417                 page(pager.getIndex() + 1);
418               },
419               platform: config.swipePlatforms
420             }
421           );
422         } else if (config.orientation == VERTICAL) {
423           listContainer.onoPagerSwipe(
424             {
425               dragDown: function() {
426                 page(pager.getIndex() - 1);
427               },
428               dragUp: function() {
429                 page(pager.getIndex() + 1);
430               },
431               platform: config.swipePlatforms
432             }
433           );
434         }
435       }
436     }

438     function page(arg_newIndex) {
439       if (config.autoPage.active == true) {
440         pager.resetAutopager();
441       }
442       var oldIndex = pager.getIndex();
443       var newIndex = pager.setIndex(arg_newIndex);
444       animation._page(oldIndex, newIndex);
445     }

491   /**
492    * Handles swipe events on mobile phones and desktops. The code is based on
493    * the QuickGestures jQuery plugin of Anders Zakrisson.
494    *
495    * @return {jQuery} chainable jQuery class.
496    * @memberOf jQuery.fn
497    * @param {Object|Null} arg_config Swipe config.
498    * @param {Function} arg_config.dragLeft Handles drag left event.
499    * @param {Function} arg_config.dragRight Handles drag right event.
500    * @param {Function} arg_config.dragUp Handles drag up event.
501    * @param {Function} arg_config.dragDown Handles drag down event.
502    * @param {Number} arg_config.threshold Number of pixels that have to be
503    *    dragged before something is considered a swipe.
504    * @param {String} arg_config.platform Determines on what platforms
505    *    the user is able to page by swiping. 'touch' activates swiping only on
506    *    touch devices. 'all' will activates swiping on touch devices and
507    *    desktop. Swiping on the desktop is done with mouse gestures. Default
508    *    value is 'touch'.
509    */
510   $.fn.onoPagerSwipe = function(arg_config) {
511     var config = {
512       dragLeft: null,
513       dragRight: null,
514       dragUp: null,
515       dragDown: null,
516       threshold: 75,
517       platform: 'touch'
518     };
519     config = $.extend(true, config, arg_config);
520 
521     return this.each(function() {
522       var data = {
523         x: 0,
524         y: 0,
525         t: null,
526         time: null
527       };
528 
529       if (config.platform == 'touch') {
530         // Check if browser supports touch events
531         var el = document.createElement('div');
532         el.setAttribute('ontouchstart', '');
533         el.setAttribute('ontouchmove', '');
534         el.setAttribute('ontouchend', '');
535         if (typeof(el.ontouchstart) != 'function' ||
536             typeof(el.ontouchmove) != 'function' ||
537             typeof(el.ontouchend) != 'function') {
538           config.platform = '';
539         }
540       }
541 
542       if (config.platform == 'touch' || config.platform == 'all') {
543         // Handle swipes on mobile browsers
544         var lastPageX, lastPageY, offsetLeft, offsetTop;
545 
546         this.addEventListener(
547           'touchstart',
548           function(e) {
549             e.preventDefault();
550             offsetLeft = ($(window).width() - $(this).outerWidth(true)) / 2;
551             offsetTop = ($(window).height() - $(this).outerHeight(true)) / 2;
552             data.x = e.targetTouches[0].pageX - offsetLeft;
553             data.y = e.targetTouches[0].pageY - offsetTop;
554             data.time = new Date();
555           },
556           false
557         );
558 
559         this.addEventListener(
560           'touchmove',
561           function(e) {
562             e.preventDefault();
563             //e.stopPropagation();
564             lastPageX = e.targetTouches[0].pageX - offsetLeft;
565             lastPageY = e.targetTouches[0].pageY - offsetTop;
566           },
567           false
568         );
569 
570         this.addEventListener(
571           'touchend',
572           function(e) {
573             e.preventDefault();
574             var diffX = lastPageX - data.x;
575             var diffY = lastPageY - data.y;
576             if (data.t != null) {
577               // End timer for hold event if it hasn't been triggered
578               clearTimeout(data.t);
579             }
580 
581             if (diffX <= -config.threshold) {
582               if ($.isFunction(config.dragLeft)) {
583                 config.dragLeft();
584               }
585             } else if (diffX >= config.threshold) {
586               if ($.isFunction(config.dragRight)) {
587                 config.dragRight();
588               }
589             }
590 
591             if (diffY <= -config.threshold) {
592               if ($.isFunction(config.dragUp)) {
593                 config.dragUp();
594               }
595             } else if (diffY >= config.threshold) {
596               if ($.isFunction(config.dragDown)) {
597                 config.dragDown();
598               }
599             }
600           },
601           false
602         );
603       }
604       if (config.platform == 'all') {
605         // Handles swipes on desktop
606         $(this).mousedown(function(e) {
607           var offsetLeft = ($(window).width() - $(this).outerWidth(true)) / 2;
608           var offsetTop = ($(window).height() - $(this).outerHeight(true)) / 2;
609           data.x = e.pageX - offsetLeft;
610           data.y = e.pageY - offsetTop;
611           data.time = new Date();
612 
613           $(this).bind('mousemove', function(e) {
614             $(this).bind('mouseup', function() {
615               $(this).unbind('mousemove');
616             });
617 
618             // Handles horizontal swipes
619             var diffX = (e.pageX - offsetLeft) - data.x;
620             var diffY = (e.pageY - offsetTop) - data.y;
621 
622             if (diffX <= -config.threshold) {
623               $(this).unbind('mousemove');
624               if ($.isFunction(config.dragLeft)) {
625                 config.dragLeft();
626               }
627             } else if (diffX >= config.threshold) {
628               $(this).unbind('mousemove');
629               if ($.isFunction(config.dragRight)) {
630                 config.dragRight();
631               }
632             }
633 
634             // Handle vertical swipes
635             if (diffY <= -config.threshold) {
636               $(this).unbind('mousemove');
637               if ($.isFunction(config.dragUp)) {
638                 config.dragUp();
639               }
640             } else if (diffY >= config.threshold) {
641               $(this).unbind('mousemove');
642               if ($.isFunction(config.dragDown)) {
643                 config.dragDown();
644               }
645             }
646           });
647         });
648       }
649     });
650   };
651 })(jQuery);

668 /**
669  * @namespace Manages the scroller control.
670  *
671  * @constructor
672  * @param {Object} arg_pageScroller The scroller, wrapped in a jQuery object.
673  * @param {Object} arg_listContainer The element that contains the list, wrapped
674  *    in a jQuery object.
675  * @param {Object} arg_list The element that contains the list, wrapped
676  *    in a jQuery object.
677  * @param {String} arg_orientation Determines on what axis the animation moves.
678  * Possible values are 'horizontal' and 'vertical' though it's possible to use
679  * other values as long as the animation object supports that value. Default
680  * value is 'horizontal'.
681  */
682 onoPager.scroller = function(arg_pageScroller,
683                              arg_listContainer,
684                              arg_list,
685                              arg_orientation) {
686   var pageScroller = arg_pageScroller;
687   var pageHandle = pageScroller.find('> div.onoPager_scrollerHandle');
688   var listContainer = arg_listContainer;
689   var list = arg_list;
690   var totalScroll;
691   var updateInterval;
692   var orientation = arg_orientation;
693   var noChangeCount = 0;
694   var previousOffset = 0;
695   var tools = onoPager.tools;
696   var listContainerSize = 0;
697   var listScrollSize = 0;
698   var interval = 10;
699   var topLeft = tools.getTopLeft(orientation);
700 
701   function _updateHandle() {
702     var positionList = tools.getOffset(orientation, list);
703     var positionListContainer = tools.getOffset(orientation, listContainer);
704     var offset = -(positionList - positionListContainer);
705     clearIntervalIfNoChange(offset);
706     var scrollPercentage = Math.round((offset / listScrollSize) * 100);
707     pageHandle.css(topLeft, ((totalScroll / 100) * scrollPercentage) + 'px');
708   }
709 
710   function clearIntervalIfNoChange(arg_offset) {
711     if (arg_offset == previousOffset) {
712       noChangeCount++;
713       if (noChangeCount > 10) {
714         clearInterval(updateInterval);
715       }
716     } else {
717       noChangeCount = 0;

1633 /**
1634  * @namespace Animation object. Moves through the list item in a linear manner.
1635  *    It also slowly scrolls when the user hovers over the "previous" or
1636  *    "next"-button.
1637  * @namespace
1638  * @param {Object} newConfig Standard configuration object.
1639  * @param {Object|Null} extraConfig Optional extra configuration object.
1640  * @return {Object} instance of an animation object.
1641  */
1642 onoPager.animation.linearScroller = function(newConfig, extraConfig) {
1643   /**
1644    * New animation object.
1645    * @memberOf onoPager.animation.linearScroller
1646    * @this
1647    */
1648   var linearScrollerInstance = new onoPager.animation._standard(
1649     newConfig, extraConfig
1650   );
1651   var tools = onoPager.tools;
1652 
1653   function checkBounds(currentMargin, list, move) {
1654     var pagePrevious = linearScrollerInstance._config.pagePrevious;
1655     var pageNext = linearScrollerInstance._config.pageNext;
1656     var listSize = tools.getInnerSize(
1657       linearScrollerInstance._config.orientation,
1658       jQuery(list)
1659     );
1660     var viewportSize = tools.getInnerSize(
1661       linearScrollerInstance._config.orientation,
1662       jQuery(list).parent()
1663     );
1664     var maxMarginBound = listSize - viewportSize;
1665     var newMargin = currentMargin - move;
1666     var DISABLED = 'disabled';
1667     
1668     if (newMargin >= 0) {
1669       pagePrevious.addClass(DISABLED);
1670       return false;
1671     } else if ((newMargin - 1) <= (-maxMarginBound)) {
1672       pageNext.addClass(DISABLED);
1673       return false;
1674     } else {
1675       pagePrevious.removeClass(DISABLED);
1676       pageNext.removeClass(DISABLED);
1677       return true;
1678     }
1679   }
1680   
1681   function currentOffset() {
1682     var topLeft = tools.getTopLeft(linearScrollerInstance._config.orientation);
1683     var currentMargin = linearScrollerInstance._config.list.css(
1684                           'margin-' + topLeft);
1685     currentMargin = currentMargin.replace('px', '');
1686     currentMargin = parseInt(currentMargin);
1687     return currentMargin;
1688   }
1689 
1690   /**
1691    * @see onoPager.animation._standard#init
1692    * @memberOf onoPager.animation.linearScroller
1693    * @this
1694    */
1695   linearScrollerInstance.init = function() {
1696     var listSize = 0;
1697     if (this._config.orientation == 'horizontal') {
1698       this._config.listItems.css('float', 'left');
1699       this._config.listItems.each(function(index) {
1700         listSize += jQuery(this).outerWidth(true);
1701       });
1702       this._config.list.css(
1703         {
1704           'width': listSize + 'px',
1705           'position': 'relative'
1706         }
1707       );
1708     this._config.listContainer.css('position', 'relative');
1709     }
1710     var viewport = tools.getInnerSize(
1711       this._config.orientation,
1712       this._config.list.parent()
1713     );
1714 
1715     if (listSize > viewport) {
1716       this._config.pageNext.show();
1717     }
1718     
1719     checkBounds(currentOffset(), this._config.list, 0)
1720   }

1732    * @see onoPager.animation._standard#pagerHover
1733    * @memberOf onoPager.animation.linearScroller
1734    * @this
1735    */
1736   linearScrollerInstance.pagerHover = function(move) {
1737     var speed = 5;
1738     if (move != 0) {
1739       var list = this._config.list;
1740       this._moveInterval = setInterval(
1741         function() {
1742           var currentMargin = currentOffset();
1743           if (!checkBounds(currentMargin, list, move)) {
1744             return;
1745           }
1746           list.css('margin-' + topLeft, (-move) + currentMargin + 'px');
1747         },
1748         speed
1749       );
1750     } else {
1751       clearInterval(this._moveInterval);
1752     }
1753   }
