1 /**
  2  * @namespace Handles animation that gives a time indication of the intervals
  3  *    between paging.
  4  */
  5 onoPager.autopageAnimation = (function() {
  6   // Throws an error if the created animation object does not comply to the
  7   // interface.
  8   function interfaceCheck(animation, publicMethods) {
  9     var hasMethod;
 10     for (var i = 0; i < publicMethods.length; i++) {
 11       hasMethod = false;
 12       for (object in animation) {
 13         if (object == publicMethods[i]) {
 14           hasMethod = true;
 15         }
 16       }
 17       if (hasMethod == false) {
 18         throw new Error('Animation object does not implement ' +
 19           'public method "' + publicMethods[i] + '"');
 20       }
 21       if (typeof(animation[object]) != 'function') {
 22         throw new Error('animation.' + publicMethods[i] + ' is not of type' +
 23           '"function", but "' + typeof(animation[object]) + '"');
 24       }
 25     }
 26   }
 27 
 28   return {
 29     /**
 30      * This method creates and returns an animation object.
 31      *
 32      * @param {Object} config Configuration object.
 33      * @return {object} The animation object.
 34      */
 35     createAnimation: function(config, extraConfig) {
 36       if (typeof(onoPager.autopageAnimation[config.autoPageAnimationType]) !=
 37           'function') {
 38         throw new Error('autoPageAnimationType "' +
 39           config.autoPageAnimationType + '" is not of type function, but ' +
 40           typeof(onoPager.autopageAnimation[config.autoPageAnimationType])
 41         );
 42       }
 43       config.root.addClass('onoPager_autopageAnimation_' +
 44         config.autoPageAnimationType);
 45 
 46       var animation = onoPager.autopageAnimation[
 47         config.autoPageAnimationType](config, extraConfig);
 48 
 49       interfaceCheck(
 50         animation,
 51         [
 52           'init',
 53           'start'
 54         ]
 55       );
 56 
 57       return animation;
 58     }
 59   };
 60 })();

 67 /**
 68  * @namespace Base component of all autopage animation objects. All properties
 69  * and methods defined here are available to all animation objects. All private
 70  * properties and methods are designated by an underscore prefix.
 71  *
 72  * @constructor
 73  * @param {Object} newConfig Standard configuration object.
 74  * @param {Object} newConfig.listContainer jQuery object. References to
 75  *    the element that is the container for the list from which the pager is
 76  *    built.
 77  * @param {Number} newConfig.animationSpeed Determines the speed at
 78  *    which the animations take place. Time value is set in milliseconds.
 79  * @param {String} newConfig.orientation Determines on what axis the
 80  *    animation moves. Possible values are 'horizontal' and 'vertical' though
 81  *    it's possible to use other values as long as the animation object
 82  *    supports that value. Default value is 'horizontal'.
 83  * @param {Object} newConfig.root The DOM element in which all animation must
 84  *    take place.
 85  * @param {Number} newConfig.autoPageInterval The interval between
 86  *    autopaging. Time value is set in milliseconds.
 87  */
 88 onoPager.autopageAnimation._standard = function(newConfig) {
 89   this._config = {
 90     listContainer: null, /* The element of the list */
 91     animationSpeed: 0, /* Time the animation takes */
 92     orientation: null, /* 'horizontal' or 'vertical' */
 93     root: null, /* The DOM element in which all animation must take place. */
 94     autoPageInterval: 0 /* The interval between autopaging */
 95   };
 96   jQuery.extend(true, this._config, newConfig);
 97 
 98   /**
 99    * This method is run when the animation object is initialized. This method
100    * is not implemented in the base animation object, but is part of the
101    * interface. Failing to implement this method will result in an error.
102    */
103   this.init = function() {}
104   delete this.init;
105 
106   this._init = function() {
107     this.init();
108   }
109 
110   /**
111    * Starts the page animation. This method is not implemented in the base
112    * animation object, but is part of the interface. Failing to implement this
113    * method will result in an error.
114    */
115   this.start = function() {}
116   delete this.page;
117 
118   this._start = function() {
119     this.start();
120   }
121 };

128 /**
129  * @namespace Autopage animation object. This object will create a simple
130  * timeline
131  *
132  * @param {Object} newConfig Standard configuration object.
133  * @return {Object} instance of an animation object.
134  */
135 onoPager.autopageAnimation.timeline = function(newConfig) {
136   /**
137    * New animation object.
138    * @memberOf onoPager.autopageAnimation.timeline
139    */
140   var timelineInstance = new onoPager.autopageAnimation._standard(newConfig);
141   var tools = onoPager.tools;
142   var bar = jQuery([]);
143   var root = timelineInstance._config.root;
144   var listContainer = timelineInstance._config.listContainer;
145 
146   /**
147    * @see onoPager.autopageAnimation._standard#init
148    * @memberOf onoPager.autopageAnimation.timeline
149    * @this
150    */
151   timelineInstance.init = function() {
152     root.html('<div class="onoPager_autoPageBar"></div>');
153     bar = root.find('div.onoPager_autoPageBar');
154     root.css(
155       {
156         width: listContainer.innerWidth(),
157         position: 'absolute',
158         /* left: listContainer.position().left + 'px', */
159         top: listContainer.position().top +
160                listContainer.innerHeight(true) + 'px'
161       }
162     );
163 
164     bar.css('width', 0);
165     bar.animate(
166       {
167         width: root.innerWidth() + 'px'
168       },
169       {
170         duration: this._config.autoPageInterval,
171         easing: 'linear'
172       }
173     );
174   }
175 
176   /**
177    * @see onoPager.autopageAnimation._standard#start
178    * @memberOf onoPager.autopageAnimation.timeline
179    * @this
180    */
181   timelineInstance.start = function() {
182     var interval = this._config.autoPageInterval - this._config.animationSpeed;
183 
184     bar.stop(true, true);
185     bar.css('width', 0);
186     bar.delay(this._config.animationSpeed).animate(
187       {
188         width: root.innerWidth() + 'px'
189       },
190       {
191         duration: interval,
192         easing: 'linear'
193       }
194     );
195   }
196 
197   return timelineInstance;
198 };

205 /**
206  * @namespace Autopage animation object. This object will create a pie chart
207  * representing a clock.
208  *
209  * @param {Object} newConfig Standard configuration object.
210  * @param {Object} arg_extraConfig Extra configuration options that are
211  *    specific to this animation object.
212  * @param {Number} arg_extraConfig.widthHeight The width and height of the
213  *    clock.
214  * @param {String} arg_extraConfig.color The hex color of the clock.
215  * @param {Number} arg_extraConfig.shadowBlur The blur range of the shadow in
216  *    pixels. Default value is 5.
217  * @param {Number} arg_extraConfig.shadowOffsetX The X-axis offset of the
218  *    shadow. A positive value will cast a shadow to the right, a negative to
219  *    the left. Default value is 2.
220  * @param {Number} arg_extraConfig.shadowOffsetY The Y-axis offset of the
221  *    shadow. A positive value will cast a shadow above the clock, a negative
222  *    one below. Default value is 2.
223  * @param {String} arg_extraConfig.shadowBackgroundColor The hex color of the
224  *    shadow.
225  * @param {Number} arg_extraConfig.intervalPrecision Determines the number of
226  *    frames in the animation. A value of 1 will give the maximum smoothness
227  *    to the animation, but the clock might then not be able to keep up with
228  *    the pager's speed. A higher number will create a more responsive
229  *    animation, but also creates less frames.. The default value is 4.
230  * @return {Object} instance of an animation object.
231  */
232 onoPager.autopageAnimation.clock = function(newConfig, arg_extraConfig) {
233   /**
234    * New animation object.
235    * @memberOf onoPager.autopageAnimation.clock
236    */
237   var clockInstance = new onoPager.autopageAnimation._standard(newConfig);
238   var extraConfig = {
239     widthHeight: 16,
240     color: '#ffffff',
241     shadowBlur: 5,
242     shadowOffsetX: 2,
243     shadowOffsetY: 2,
244     shadowBackgroundColor: '#999999',
245     intervalPrecision: 4
246   };
247   jQuery.extend(true, extraConfig, arg_extraConfig);
248   var canvasWidth = extraConfig.widthHeight + extraConfig.shadowBlur;
249   if (extraConfig.shadowOffsetX < 0) {
250     canvasWidth += -extraConfig.shadowOffsetX;
251   } else {
252     canvasWidth += extraConfig.shadowOffsetX;
253   }
254 
255   var canvasHeight = extraConfig.widthHeight + extraConfig.shadowBlur;
256   if (extraConfig.shadowOffsetY < 0) {
257     canvasHeight += -extraConfig.shadowOffsetY;
258   } else {
259     canvasHeight += extraConfig.shadowOffsetY;
260   }
261 
262   var degrees;
263   var root = clockInstance._config.root;
264   var canvas;
265   var context;
266   var drawClockInterval;
267   var drawClockTimeout;
268   var listContainer = clockInstance._config.listContainer;
269   var interval;
270 
271   function drawClock() {
272       degrees += extraConfig.intervalPrecision;
273 
274       var centerX = Math.floor((extraConfig.widthHeight / 2) +
275                                extraConfig.shadowOffsetX);
276       var centerY = Math.floor((extraConfig.widthHeight / 2) +
277                                extraConfig.shadowOffsetY);
278       var radius = Math.floor(extraConfig.widthHeight / 2);
279 
280       context.clearRect(0, 0, canvasWidth + 10, canvasHeight + 10);
281       context.beginPath();
282       context.moveTo(centerX, centerY);
283       context.arc(centerX,
284                   centerY,
285                   radius,
286                   (Math.PI / 180) * -90,
287                   (Math.PI / 180) * degrees,
288                   false);
289       context.lineTo(centerX, centerY);
290       context.closePath();
291 
292       context.fillStyle = extraConfig.color;
293       context.shadowColor = extraConfig.shadowBackgroundColor;
294       context.shadowBlur = extraConfig.shadowBlur;
295       context.shadowOffsetX = extraConfig.shadowOffsetX;
296       context.shadowOffsetY = extraConfig.shadowOffsetY;
297       context.fill();
298   };
299 
300   /**
301    * @see onoPager.autopageAnimation._standard#init
302    * @memberOf onoPager.autopageAnimation.timeline
303    * @this
304    */
305   clockInstance.init = function() {
306     interval = this._config.autoPageInterval - this._config.animationSpeed;
307     root.html('<canvas width="' + canvasWidth + '" height="' +
308               canvasHeight + '"></canvas>');
309     canvas = root.find('canvas')[0];
310     context = canvas.getContext('2d');
311     this.start();
312   }
313 
314   /**
315    * @see onoPager.autopageAnimation._standard#start
316    * @memberOf onoPager.autopageAnimation.timeline
317    * @this
318    */
319   clockInstance.start = function() {
320     clearInterval(drawClockInterval);
321     degrees = -80;
322     context.clearRect(0, 0, canvasWidth, canvasHeight);
323     drawClockTimeout = setTimeout(
324       function() {
325         drawClockInterval = setInterval(
326           drawClock,
327           Math.floor(interval / (360 / extraConfig.intervalPrecision))
328         );
329       },
330       this._config.animationSpeed);
331   }
332 
333   return clockInstance;
334 };
