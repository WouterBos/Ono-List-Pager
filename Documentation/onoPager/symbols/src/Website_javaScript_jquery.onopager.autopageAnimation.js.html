 11 /**
 12  * @namespace Handles animation that gives a time indication of the intervals
 13  *    between paging.
 14  */
 15 onoPager.autopageAnimation = (function() {
 16   // Throws an error if the created animation object does not comply to the
 17   // interface.
 18   function interfaceCheck(animation, publicMethods) {
 19     var hasMethod;
 20     for (var i = 0; i < publicMethods.length; i++) {
 21       hasMethod = false;
 22       for (object in animation) {
 23         if (object == publicMethods[i]) {
 24           hasMethod = true;
 25         }
 26       }
 27       if (hasMethod == false) {
 28         throw new Error('Animation object does not implement ' +
 29           'public method "' + publicMethods[i] + '"');
 30       }
 31       if (typeof(animation[object]) != 'function') {
 32         throw new Error('animation.' + publicMethods[i] + ' is not of type' +
 33           '"function", but "' + typeof(animation[object]) + '"');
 34       }
 35     }
 36   }
 37 
 38   return {
 39     /**
 40      * This method creates and returns an animation object.
 41      *
 42      * @param {Object} config Configuration object.
 43      * @return {object} The animation object.
 44      */
 45     createAnimation: function(config, extraConfig) {
 46       if (typeof(onoPager.autopageAnimation[config.autoPageAnimationType]) !=
 47           'function') {
 48         throw new Error('autoPageAnimationType "' +
 49           config.autoPageAnimationType + '" is not of type function, but ' +
 50           typeof(onoPager.autopageAnimation[config.autoPageAnimationType])
 51         );
 52       }
 53 
 54       var animation;
 55 
 56       config.root.addClass('onoPager_autopageAnimation_' +
 57         config.autoPageAnimationType);
 58 
 59       var animationSupport = true;
 60       if ('isSupportedByBrowser' in
 61           onoPager.autopageAnimation[config.autoPageAnimationType]) {
 62         if (onoPager.autopageAnimation[config.autoPageAnimationType]
 63             .isSupportedByBrowser() == false) {
 64           animationSupport = false;
 65         }
 66       }
 67 
 68       if (animationSupport == true) {
 69         animation = onoPager.autopageAnimation[
 70           config.autoPageAnimationType](config, extraConfig);
 71         interfaceCheck(
 72           animation,
 73           [
 74             'init',
 75             'start'
 76           ]
 77         );
 78       }
 79 
 80       return animation;
 81     }
 82   };
 83 })();

 90 /**
 91  * @namespace Base component of all autopage animation objects. All properties
 92  * and methods defined here are available to all animation objects. All private
 93  * properties and methods are designated by an underscore prefix.
 94  *
 95  * @constructor
 96  * @param {Object} newConfig Standard configuration object.
 97  * @param {Object} newConfig.listContainer jQuery object. References to
 98  *    the element that is the container for the list from which the pager is
 99  *    built.
100  * @param {Number} newConfig.animationSpeed Determines the speed at
101  *    which the animations take place. Time value is set in milliseconds.
102  * @param {String} newConfig.orientation Determines on what axis the
103  *    animation moves. Possible values are 'horizontal' and 'vertical' though
104  *    it's possible to use other values as long as the animation object
105  *    supports that value. Default value is 'horizontal'.
106  * @param {Object} newConfig.root The DOM element in which all animation must
107  *    take place.
108  * @param {Number} newConfig.autoPageInterval The interval between
109  *    autopaging. Time value is set in milliseconds.
110  */
111 onoPager.autopageAnimation._standard = function(newConfig) {
112   this._config = {
113     listContainer: null, /* The element of the list */
114     animationSpeed: 0, /* Time the animation takes */
115     orientation: null, /* 'horizontal' or 'vertical' */
116     root: null, /* The DOM element in which all animation must take place. */
117     autoPageInterval: 0 /* The interval between autopaging */
118   };
119   jQuery.extend(true, this._config, newConfig);
120 
121   /**
122    * This method is run when the animation object is initialized. This method
123    * is not implemented in the base animation object, but is part of the
124    * interface. Failing to implement this method will result in an error.
125    */
126   this.init = function() {}
127   delete this.init;
128 
129   this._init = function() {
130     this.init();
131   }
132 
133   /**
134    * Starts the page animation. This method is not implemented in the base
135    * animation object, but is part of the interface. Failing to implement this
136    * method will result in an error.
137    */
138   this.start = function() {}
139   delete this.page;
140 
141   this._start = function() {
142     this.start();
143   }
144 };

151 /**
152  * @namespace Autopage animation object. This object will create a simple
153  * timeline.
154  *
155  * @param {Object} newConfig Standard configuration object.
156  * @return {Object} instance of an animation object.
157  */
158 onoPager.autopageAnimation.timeline = function(newConfig) {
159   /**
160    * New animation object.
161    * @memberOf onoPager.autopageAnimation.timeline
162    */
163   var timelineInstance = new onoPager.autopageAnimation._standard(newConfig);
164   var tools = onoPager.tools;
165   var bar = jQuery([]);
166   var root = timelineInstance._config.root;
167   var listContainer = timelineInstance._config.listContainer;
168   var startTimeout;
169 
170   /**
171    * @see onoPager.autopageAnimation._standard#init
172    * @memberOf onoPager.autopageAnimation.timeline
173    * @this
174    */
175   timelineInstance.init = function() {
176     // Setup pager markup and reference.
177     root.html('<div class="onoPager_autoPageBar"></div>');
178     bar = root.find('div.onoPager_autoPageBar');
179 
180     // Setup styling
181     root.css(
182       {
183         width: listContainer.innerWidth(),
184         position: 'absolute',
185         top: listContainer.position().top +
186                listContainer.innerHeight(true) + 'px'
187       }
188     );
189     bar.css('width', 0);
190 
191     // Run animation
192     bar.animate(
193       {
194         width: root.innerWidth() + 'px'
195       },
196       {
197         duration: this._config.autoPageInterval,
198         easing: 'linear'
199       }
200     );
201   }
202 
203   /**
204    * @see onoPager.autopageAnimation._standard#start
205    * @memberOf onoPager.autopageAnimation.timeline
206    * @this
207    */
208   timelineInstance.start = function() {
209     var intervalTime = this._config.autoPageInterval -
210                        this._config.animationSpeed; // interval time minus
211                                                     // transition time
212     var animationSpeed = this._config.animationSpeed;
213 
214     bar.stop(true, true); // End all current animations
215     clearTimeout(startTimeout); // Remove any pending animations
216     bar.css('width', 0); // Reset the timeline bar
217 
218     // Schedule start for timeline animation
219     startTimeout = setTimeout(
220       function() {
221         bar.animate(
222           {
223             width: root.innerWidth() + 'px'
224           },
225           {
226             duration: intervalTime,
227             easing: 'linear'
228           }
229         );
230       },
231       animationSpeed
232     );
233   }
234 
235   return timelineInstance;
236 };

243 /**
244  * @namespace Autopage animation object. This object will create a pie chart
245  * representing a clock.
246  *
247  * @param {Object} newConfig Standard configuration object.
248  * @param {Object} arg_extraConfig Extra configuration options that are
249  *    specific to this animation object.
250  * @param {Number} arg_extraConfig.widthHeight The width and height of the
251  *    clock.
252  * @param {String} arg_extraConfig.color The hex color of the clock.
253  * @param {Number} arg_extraConfig.shadowBlur The blur range of the shadow in
254  *    pixels. Default value is 5.
255  * @param {Number} arg_extraConfig.shadowOffsetX The X-axis offset of the
256  *    shadow. A positive value will cast a shadow to the right, a negative to
257  *    the left. Default value is 2.
258  * @param {Number} arg_extraConfig.shadowOffsetY The Y-axis offset of the
259  *    shadow. A positive value will cast a shadow above the clock, a negative
260  *    one below. Default value is 2.
261  * @param {String} arg_extraConfig.shadowBackgroundColor The hex color of the
262  *    shadow.
263  * @param {Number} arg_extraConfig.intervalPrecision Determines the number of
264  *    frames in the animation. A value of 1 will give the maximum smoothness
265  *    to the animation, but the clock might then not be able to keep up with
266  *    the pager's speed. A higher number will create a more responsive
267  *    animation, but also creates less frames.. The default value is 4.
268  * @return {Object} instance of an animation object.
269  */
270 onoPager.autopageAnimation.clock = function(newConfig, arg_extraConfig) {
271   /**
272    * New animation object.
273    * @memberOf onoPager.autopageAnimation.clock
274    */
275   var clockInstance = new onoPager.autopageAnimation._standard(newConfig);
276   var extraConfig = {
277     widthHeight: 16,
278     color: '#ffffff',
279     shadowBlur: 5,
280     shadowOffsetX: 2,
281     shadowOffsetY: 2,
282     shadowBackgroundColor: '#999999',
283     intervalPrecision: 4
284   };
285   jQuery.extend(true, extraConfig, arg_extraConfig);
286   var canvasWidth = extraConfig.widthHeight + extraConfig.shadowBlur;
287   if (extraConfig.shadowOffsetX < 0) {
288     canvasWidth += -extraConfig.shadowOffsetX;
289   } else {
290     canvasWidth += extraConfig.shadowOffsetX;
291   }
292 
293   var canvasHeight = extraConfig.widthHeight + extraConfig.shadowBlur;
294   if (extraConfig.shadowOffsetY < 0) {
295     canvasHeight += -extraConfig.shadowOffsetY;
296   } else {
297     canvasHeight += extraConfig.shadowOffsetY;
298   }
299 
300   var degrees;
301   var root = clockInstance._config.root;
302   var canvas;
303   var context;
304   var drawClockInterval;
305   var drawClockTimeout;
306   var listContainer = clockInstance._config.listContainer;
307   var intervalTime;
308 
309   function drawClock() {
310       degrees += extraConfig.intervalPrecision;
311 
312       var centerX = Math.floor((extraConfig.widthHeight / 2) +
313                                extraConfig.shadowOffsetX);
314       var centerY = Math.floor((extraConfig.widthHeight / 2) +
315                                extraConfig.shadowOffsetY);
316       var radius = Math.floor(extraConfig.widthHeight / 2);
317 
318       context.clearRect(0, 0, canvasWidth + 10, canvasHeight + 10);
319       context.beginPath();
320       context.moveTo(centerX, centerY);
321       context.arc(centerX,
322                   centerY,
323                   radius,
324                   (Math.PI / 180) * -90,
325                   (Math.PI / 180) * degrees,
326                   false);
327       context.lineTo(centerX, centerY);
328       context.closePath();
329 
330       context.fillStyle = extraConfig.color;
331       context.shadowColor = extraConfig.shadowBackgroundColor;
332       context.shadowBlur = extraConfig.shadowBlur;
333       context.shadowOffsetX = extraConfig.shadowOffsetX;
334       context.shadowOffsetY = extraConfig.shadowOffsetY;
335       context.fill();
336   };
337 
338   /**
339    * @see onoPager.autopageAnimation._standard#init
340    * @memberOf onoPager.autopageAnimation.timeline
341    * @this
342    */
343   clockInstance.init = function() {
344     root.html('<canvas width="' + canvasWidth + '" height="' +
345               canvasHeight + '"></canvas>');
346     canvas = root.find('canvas')[0];
347     context = canvas.getContext('2d');
348     this.start(this._config.autoPageInterval);
349   }
350 
351   /**
352    * @see onoPager.autopageAnimation._standard#start
353    * @memberOf onoPager.autopageAnimation.timeline
354    * @this
355    */
356   clockInstance.start = function(autoPageInterval) {
357     var intervalTime;
358     var animationSpeed;
359     if (autoPageInterval) {
360       intervalTime = autoPageInterval;
361       animationSpeed = 0;
362     } else {
363       intervalTime = this._config.autoPageInterval -
364                          this._config.animationSpeed; // interval time minus
365                                                       // transition time
366       animationSpeed = this._config.animationSpeed;
367     }
368     clearInterval(drawClockInterval);
369     degrees = -80;
370     context.clearRect(0, 0, canvasWidth, canvasHeight);
371     drawClockTimeout = setTimeout(
372       function() {
373         drawClockInterval = setInterval(
374           drawClock,
375           Math.floor(intervalTime / (360 / extraConfig.intervalPrecision))
376         );
377       },
378       animationSpeed);
379   }
380 
381   return clockInstance;
382 };

385 /**
386  * @namespace Checks browser support for the clock animation object.
387  *
388  * @param {Object} config Configuration object.
389  * @return {object} The animation object.
390  */
391 onoPager.autopageAnimation.clock.isSupportedByBrowser = function() {
392   var canvas = document.createElement('canvas');
393   if (canvas.getContext) {
394     return true;
395   } else {
396     return false;
397   }
