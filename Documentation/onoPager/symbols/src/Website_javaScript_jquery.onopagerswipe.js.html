  3  *               QuickGestures jQuery plugin of Anders Zakrisson.
  4  */

 11 (function($) {
 12   /**
 13    * Handles swipe events on mobile phones and desktops. The code is based on
 14    * the QuickGestures jQuery plugin of Anders Zakrisson.
 15    *
 16    * @return {Object} chainable jQuery class.
 17    * @memberOf jQuery.fn
 18    * @param {Object|Null} arg_config Swipe config.
 19    * @param {Function} arg_config.dragLeft Handles drag left event.
 20    * @param {Function} arg_config.dragRight Handles drag right event.
 21    * @param {Function} arg_config.dragUp Handles drag up event.
 22    * @param {Function} arg_config.dragDown Handles drag down event.
 23    * @param {Number} arg_config.threshold Number of pixels that have to be
 24    *    dragged before something is considered a swipe.
 25    * @param {String} arg_config.platform Determines on what platforms
 26    *    the user is able to page by swiping. 'touch' activates swiping only on
 27    *    touch devices. 'all' will activates swiping on touch devices and
 28    *    desktop. Swiping on the desktop is done with mouse gestures. Default
 29    *    value is 'touch'.
 30    */
 31   jQuery.fn.onoPagerSwipe = function(arg_config) {
 32     var config = {
 33       dragLeft: null,
 34       dragRight: null,
 35       dragUp: null,
 36       dragDown: null,
 37       threshold: 75,
 38       platform: 'touch'
 39     };
 40     config = $.extend(true, config, arg_config);
 41 
 42     return this.each(function() {
 43       var data = {
 44         x: 0,
 45         y: 0,
 46         t: null,
 47         time: null
 48       };
 49 
 50       if (config.platform == 'touch') {
 51         // Check if browser supports touch events
 52         var el = document.createElement('div');
 53         el.setAttribute('ontouchstart', '');
 54         el.setAttribute('ontouchmove', '');
 55         el.setAttribute('ontouchend', '');
 56         if (typeof(el.ontouchstart) != 'function' ||
 57             typeof(el.ontouchmove) != 'function' ||
 58             typeof(el.ontouchend) != 'function') {
 59           config.platform = '';
 60         }
 61       }
 62 
 63       if (config.platform == 'touch' || config.platform == 'all') {
 64         // Handle swipes on mobile browsers
 65         var lastPageX, lastPageY, offsetLeft, offsetTop;
 66 
 67         this.addEventListener(
 68           'touchstart',
 69           function(e) {
 70             //e.preventDefault();
 71             offsetLeft = ($(window).width() - $(this).outerWidth(true)) / 2;
 72             offsetTop = ($(window).height() - $(this).outerHeight(true)) / 2;
 73             data.x = e.targetTouches[0].pageX - offsetLeft;
 74             data.y = e.targetTouches[0].pageY - offsetTop;
 75             data.time = new Date();
 76           },
 77           false
 78         );
 79 
 80         this.addEventListener(
 81           'touchmove',
 82           function(e) {
 83             e.preventDefault();
 84             //e.stopPropagation();
 85             lastPageX = e.targetTouches[0].pageX - offsetLeft;
 86             lastPageY = e.targetTouches[0].pageY - offsetTop;
 87           },
 88           false
 89         );
 90 
 91         this.addEventListener(
 92           'touchend',
 93           function(e) {
 94             e.preventDefault();
 95             var diffX = lastPageX - data.x;
 96             var diffY = lastPageY - data.y;
 97             if (data.t != null) {
 98               // End timer for hold event if it hasn't been triggered
 99               clearTimeout(data.t);
100             }
101 
102             if (diffX <= -config.threshold) {
103               if ($.isFunction(config.dragLeft)) {
104                 config.dragLeft();
105               }
106             } else if (diffX >= config.threshold) {
107               if ($.isFunction(config.dragRight)) {
108                 config.dragRight();
109               }
110             }
111 
112             if (diffY <= -config.threshold) {
113               if ($.isFunction(config.dragUp)) {
114                 config.dragUp();
115               }
116             } else if (diffY >= config.threshold) {
117               if ($.isFunction(config.dragDown)) {
118                 config.dragDown();
119               }
120             }
121           },
122           false
123         );
124       }
125       if (config.platform == 'all') {
126         // Handles swipes on desktop
127         $(this).mousedown(function(e) {
128           var offsetLeft = ($(window).width() - $(this).outerWidth(true)) / 2;
129           var offsetTop = ($(window).height() - $(this).outerHeight(true)) / 2;
130           data.x = e.pageX - offsetLeft;
131           data.y = e.pageY - offsetTop;
132           data.time = new Date();
133 
134           $(this).bind('mousemove', function(e) {
135             $(this).bind('mouseup', function() {
136               $(this).unbind('mousemove');
137             });
138 
139             // Handles horizontal swipes
140             var diffX = (e.pageX - offsetLeft) - data.x;
141             var diffY = (e.pageY - offsetTop) - data.y;
142 
143             if (diffX <= -config.threshold) {
144               $(this).unbind('mousemove');
145               if ($.isFunction(config.dragLeft)) {
146                 config.dragLeft();
147               }
148             } else if (diffX >= config.threshold) {
149               $(this).unbind('mousemove');
150               if ($.isFunction(config.dragRight)) {
151                 config.dragRight();
152               }
153             }
154 
155             // Handle vertical swipes
156             if (diffY <= -config.threshold) {
157               $(this).unbind('mousemove');
158               if ($.isFunction(config.dragUp)) {
159                 config.dragUp();
160               }
161             } else if (diffY >= config.threshold) {
162               $(this).unbind('mousemove');
163               if ($.isFunction(config.dragDown)) {
164                 config.dragDown();
165               }
166             }
167           });
168         });
169       }
170     });
171   };
172 })(jQuery);
