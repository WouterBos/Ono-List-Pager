  2  * @namespace Animation namespace

  4 onoPager.animation = (function() {
  5   // Throws an error if the created animation object does not comply to the
  6   // interface.
  7   function interfaceCheck(animation, publicMethods) {
  8     var hasMethod;
  9     for (var i = 0; i < publicMethods.length; i++) {
 10       hasMethod = false;
 11       for (object in animation) {
 12         if (object == publicMethods[i]) {
 13           hasMethod = true;
 14         }
 15       }
 16       if (hasMethod == false) {
 17         throw new Error('Animation object does not implement ' +
 18           'public method "' + publicMethods[i] + '"');
 19       }
 20       if (typeof(animation[object]) != 'function') {
 21         throw new Error('animation.' + publicMethods[i] + ' is not of type' +
 22           '"function", but "' + typeof(animation[object]) + '"');
 23       }
 24     }
 25   }
 26 
 27   return {
 28     /**
 29      * This method creates and returns an animation object.
 30      *
 31      * @param {String} animationType Name of the animation that must be loaded.
 32      * @param {Object} config Configuration object.
 33      * @param {Object} extraConfig Extra, custom configuration object for the
 34      *   animation object.
 35      * @return {object} The animation object.
 36      */
 37     createAnimation: function(animationType, config, extraConfig) {
 38       if (typeof(onoPager.animation[animationType]) != 'function') {
 39         throw new Error('animationType "' + animationType + '" is not of ' +
 40           'type function, but ' + typeof(onoPager.animation[animationType]));
 41       }
 42       config.root.addClass('onoPager_animation_' + animationType);
 43 
 44       var animation = onoPager.animation[animationType](
 45         {
 46           list: config.list,
 47           listContainer: config.listContainer,
 48           listItems: config.listItems,
 49           animationSpeed: config.animationSpeed,
 50           orientation: config.orientation,
 51           pagePerItem: config.pagePerItem,
 52           pageNext: config.pageNext,
 53           pagePrevious: config.pagePrevious,
 54           activeIndex: config.activeIndex,
 55           animationEasing: config.animationEasing
 56         },
 57         extraConfig
 58       );
 59 
 60       interfaceCheck(
 61         animation,
 62         [
 63           'page',
 64           'init',
 65           'pagerHover'
 66         ]
 67       );
 68 
 69       return animation;
 70     }
 71   };
 72 })();

111   /**
112    * Checks if arg_scroll is not out of bounds. Use this method if you want to
113    * make sure that the list makes use of the whole space of the viewport. The
114    * argument arg_scroll is expected to be the offset of the list, like a
115    * margin. If the offset is too high or too low, it wil return the maximum
116    * or minimum scroll value. In other instances it will just return the
117    * arg_scroll value.
118    *
119    * @param {Number} arg_scroll The offset of the list in pixels.
120    * @return {Number} Returns either arg_scroll or the maximum or minimum scroll
121    * value.
122    * @this
123    */
124   this._checkMaxScroll = function(arg_scroll) {
125     var tools = onoPager.tools;
126     var orientation = this._config.orientation;
127     var listSize = tools.getInnerSize(orientation, this._config.list);
128     var listContainerSize = tools.getInnerSize(
129       orientation,
130       this._config.listContainer
131     );
132 
133     maxScroll = listSize - listContainerSize;
134     var scroll = arg_scroll;
135     if (scroll > maxScroll) {
136       scroll = maxScroll;
137     }
138     if (scroll < 0) {
139       scroll = 0;
140     }
141     return scroll;
142   }

144   /**
145    * Calculates how many times a user must page to reach the end.
146    *
147    * @return {Number} Total number of available pages. Most of the times the
148    * return value is calculated with the private function _getPagesLength().
149    * @this
150    */
151   this.getPagesLength = function() {
152     var listContainer = this._config.listContainer;
153     var listItems = this._config.listItems;
154 
155     if (this._config.pagePerItem == true) {
156       return this._config.listItems.size();
157     } else {
158       var listItemsWidth = (listItems.size() * listItems.outerWidth(true));
159       return Math.ceil(listItemsWidth / listContainer.innerWidth());
160     }
161   }

169   this._setListContainerHeight = function(listContainer, listItems) {
170     if (listItems.size() > 1) {

174           maxHeight = jQuery(this).innerHeight(true);
175         }
176       });
177       listContainer.height(maxHeight);

335   /**
336    * @see onoPager.animation._standard#page
337    * @memberOf onoPager.animation.slides
338    * @this
339    */
340   slidesInstance.page = function(oldIndex, newIndex) {
341     var oldItemLeft = 0;
342     var newItemLeft = 0;
343     var pageSize = tools.getInnerSize(
344       slidesInstance._config.orientation,
345       slidesInstance._config.listItems
346     );
347 
348     if (oldIndex < newIndex) {
349       // Next
350       oldItemLeft = pageSize;
351       newItemLeft = -pageSize;
352     }
353     if (oldIndex > newIndex) {
354       // Previous
355       oldItemLeft = -pageSize;
356       newItemLeft = pageSize;
357     }
358 
359     jQuery(this._config.listItems[oldIndex]).stop(true, true);
360     jQuery(this._config.listItems[newIndex]).stop(true, true);
361 
362     this._config.listItems.each(function() {
363       var item = jQuery(this);
364       if (item.index() != oldIndex && item.index() != newIndex) {
365         item.hide();
366       }
367     });
368 
369     var newCss = {};
370     var topLeft = tools.getTopLeft(slidesInstance._config.orientation);
371     newCss['display'] = 'block';
372     newCss[topLeft] = oldItemLeft + 'px';
373     jQuery(this._config.listItems[newIndex]).css(newCss);
374 
375     var oldAni = {};
376     oldAni[topLeft] = newItemLeft + 'px';
377     jQuery(this._config.listItems[oldIndex]).animate(
378       oldAni,
379       this._config.animationSpeed
380     );
381 
382     var newAni = {};
383     newAni[topLeft] = '0';
384     jQuery(this._config.listItems[newIndex])
385       .delay(this._config.animationSpeed / 8)
386       .animate(newAni, this._config.animationSpeed);
387   }

464         'position': 'relative'
465     });
466 
467     var offset = tools.getPosition(
468       this._config.orientation,
469       jQuery(this._config.listItems[this._config.activeIndex])
470     );
471     if (this._config.orientation == 'horizontal') {
472       this._config.list.css({ 'left': '-' + offset + 'px' });
473     } else {
474       this._config.list.css({ 'top': '-' + offset + 'px' });

487     this._config.list.stop(true, false);
488     var offset;

494     } else {
495       var size = tools.getInnerSize(

507     this._config.list.animate(
508       cssObj,
509       {
510         duration: 1000,
511         easing: this._config.animationEasing,
512         complete: function() {
513           linearInstance._setActiveClass(newIndex, true);
514         }
515       }
516     );
517   }

540  * @param {Object|Null} extraConfig Optional extra configuration object.
541  * @return {Object} instance of an animation object.
542  */
543 onoPager.animation.linearContinous = function(newConfig, extraConfig) {
544   /**
545    * New animation object.
546    * @memberOf onoPager.animation.linearContinous
547    * @this
548    */
549   var linearContinousInstance = new onoPager.animation._standard(newConfig,
550                                                                  extraConfig);
551   var tools = onoPager.tools;
552   var listItemSize = tools.getInnerSize(
553     linearContinousInstance._config.orientation,
554     linearContinousInstance._config.listItems
555   );
556   var hasCenterBackground = false;
557   var newListItems; // contains the new list of items, after duplication
558                     // takes place in onPagerCreated
559 
560   // Appends and prepends items until the list always fills the screen.
561   function fillIdleSpace(idleSpace) {
562     var list = linearContinousInstance._config.list;
563     var listItems = linearContinousInstance._config.listItems;
564 
565     fillBefore();
566     fillAfter();
567 
568     function fillBefore() {
569       // Add items to the begin of the list
570       var prependItems = jQuery([]);
571       var prependItemsArray = [];
572       var prependSpace = 0;
573 
574       for (var i = 1; i <= listItems.size(); i++) {
575         if (prependSpace < idleSpace) {
576           prependItemsArray.push(jQuery(listItems.get(-i)).clone(true));
577           if (i > 1) {
578             prependSpace += tools.getInnerSize(
579               linearContinousInstance._config.orientation,
580               jQuery(listItems.get(-i))
581             );
582           }
583         } else {
584           break;
585         }
586       }
587       for (var i = 0; i < prependItemsArray.length; i++) {
588         list.prepend(jQuery(prependItemsArray[i]).clone(true));
589       }
590     }
591 
592     function fillAfter() {
593       // Add items at the end of the list
594       var appendItems = jQuery([]);
595       var appendSpace = 0;
596 
597       for (var i = 0; i < listItems.size(); i++) {
598         if (appendSpace < idleSpace) {
599           appendItems = appendItems.add(jQuery(listItems.get(i)).clone(true));
600           if (i > 0) {
601             appendSpace += tools.getInnerSize(
602               linearContinousInstance._config.orientation,
603               jQuery(listItems.get(i))
604             );
605           }
606         } else {
607           break;
608         }
609       }
610       list.append(appendItems);
611     }
612   }
613 
614   linearContinousInstance._setListContainerHeight = function(listContainer,
615                                                              listItems) {
616     if (listItems.size() > 1) {
617       var maxHeight = 0;
618       listItems.each(function() {
619         if (maxHeight < jQuery(this).innerHeight(true)) {
620           maxHeight = jQuery(this).innerHeight(true);
621         }
622       });
623       listContainer.height(maxHeight);
624     }
625   }

651     if (oldIndex != newIndex) {
652       linearContinousInstance._setActiveClass(oldIndex, false);
653     }
654     this._config.list.stop(true, false);
655 
656     var oldItem = resetPosition(oldIndex, newIndex);
657 
658     var offset;
659     if (this._config.pagePerItem == true) {
660       offset = tools.getPosition(
661         this._config.orientation,
662         jQuery(this._config.listItems[newIndex])
663       );
664     }// else {
665     //  var size = tools.getInnerSize(

667     //    this._config.listContainer
668     //  );
669     //  offset = size * jQuery(this._config.listItems[newIndex]).index();
670     //}

674     if (hasCenterBackground) {
675       animateBackground(oldIndex, newIndex, oldItem);
676     }
677 
678     var cssObj = {};
679     var topLeft = tools.getTopLeft(this._config.orientation);
680     cssObj[topLeft] = '-' + offset + 'px';
681     this._config.list.animate(
682       cssObj,
683       {
684         duration: 1000,
685         easing: this._config.animationEasing,
686         complete: function() {
687           linearContinousInstance._setActiveClass(newIndex, true);
688         }

692     function animateBackground(oldIndex, newIndex, oldItem) {
693       var move = 0; // -1 is a move to the left, +1 is a move to the right.
694 
695       // Determine move direction
696       var maxItems = linearContinousInstance._config.listItems.size();
697       if ((oldIndex < newIndex) ||
698           (newIndex == 0 && newIndex == (maxItems - 1))) {
699         move = 1;
700       } else {
701         move = -1;
702       }
703 
704       // Animate backgrounds
705       var newItem = jQuery(linearContinousInstance._config.listItems[newIndex]);
706       var oldItem2 = jQuery(
707                        linearContinousInstance._config.listItems[oldIndex]);
708       var containerWidth =
709             linearContinousInstance._config.listContainer.innerWidth();
710       if (oldItem.index() != oldItem2.index()) {
711         oldItem2.find('*.onoPager_linearContinuous_background').css(
712           {left: containerWidth + 'px'}
713         );
714       }
715 
716       if (move > 0) {
717         console.log('forward');
718         oldItem.find('*.onoPager_linearContinuous_background').css(
719           {left: 0}
720         );
721         oldItem.find('*.onoPager_linearContinuous_background').animate(
722           {left: containerWidth + 'px'},
723           {
724             duration: 1000,
725             easing: linearContinousInstance._config.animationEasing
726           }
727         );
728         newItem.find('*.onoPager_linearContinuous_background').css(
729           {left: -containerWidth + 'px'}
730         );
731         newItem.find('*.onoPager_linearContinuous_background').animate(
732           {left: 0},
733           {
734             duration: 1000,
735             easing: linearContinousInstance._config.animationEasing
736           }
737         );
738       } else {
739         console.log('backward');
740         oldItem.find('*.onoPager_linearContinuous_background').css(
741           {left: 0}
742         );
743         oldItem.find('*.onoPager_linearContinuous_background').animate(
744           {left: -containerWidth + 'px'},
745           {
746             duration: 1000,
747             easing: linearContinousInstance._config.animationEasing
748           }
749         );
750         newItem.find('*.onoPager_linearContinuous_background').css(
751           {left: containerWidth + 'px'}
752         );
753         newItem.find('*.onoPager_linearContinuous_background').animate(
754           {left: 0},
755           {
756             duration: 1000,
757             easing: linearContinousInstance._config.animationEasing
758           }
759         );
760       }
761     }

763     // To create the appearence of a list that repeats itself infinitely, the
764     // list is repositioned just before it threatens to get out of bounds.
765     function resetPosition(oldIndex, newIndex) {
766       var listSize = linearContinousInstance._config.listItems.size();
767       var topLeft = tools.getTopLeft(
768         linearContinousInstance._config.orientation
769       );
770       var offset;
771       var oldItem;
772 
773       if (oldIndex == (listSize - 1) && newIndex == 0) {
774         // If user pages from last item to first item, position on the item
775         // *before* the first item.
776         var firstIndex = newListItems.index(
777           linearContinousInstance._config.listItems[newIndex]
778         );
779 
780         if (linearContinousInstance._config.pagePerItem == true) {
781           offset = tools.getPosition(
782             linearContinousInstance._config.orientation,
783             jQuery(newListItems[firstIndex - 1])
784           );
785           offset = Math.round(offset);
786           oldItem = jQuery(newListItems[firstIndex - 1]);
787         }// else {
788         //  var size = tools.getInnerSize(
789         //    linearContinousInstance._config.orientation,
790         //    linearContinousInstance._config.listContainer
791         //  );
792         //  offset = size * jQuery(
793         //      linearContinousInstance._config.listItems[newIndex]
794         //    ).index();
795         //}
796       } else if (oldIndex == 0 && newIndex == (listSize - 1)) {
797         // If user pages from the first item to last item, position on the item
798         // *after* the last item.
799         var lastIndex = newListItems.index(
800           linearContinousInstance._config.listItems[newIndex]
801         );
802 
803         if (linearContinousInstance._config.pagePerItem == true) {
804           offset = tools.getPosition(
805             linearContinousInstance._config.orientation,
806             jQuery(newListItems[lastIndex + 1])
807           );
808           offset = Math.round(offset);
809           oldItem = jQuery(newListItems[lastIndex + 1]);
810         }// else {
811         //  var size = tools.getInnerSize(
812         //    linearContinousInstance._config.orientation,
813         //    linearContinousInstance._config.listContainer
814         //  );
815         //  offset = size * jQuery(
816         //    linearContinousInstance._config.listItems[newIndex]
817         //  ).index();
818         //}
819       } else {
820         oldItem = jQuery(linearContinousInstance._config.listItems[oldIndex]);
821       }
822 
823       if (offset) {
824         linearContinousInstance._config.list.css(topLeft, '-' + offset + 'px');
825       }
826 
827       return oldItem;
828     }
829   }

855 
856     // Set active item
857     linearContinousInstance._setActiveClass(

859       true
860     );
861 
862     // Set list width/height
863     newListItems = this._config.list.find('> .onoPager_listItem');
864     newListItems.each(function(index) {
865       listBounds += tools.getOuterSize(
866         linearContinousInstance._config.orientation,
867         jQuery(this)

869     });
870     newListItems.css('float', 'left');
871     this._config.list.css('position', 'relative');
872     this._config.list.css(
873       tools.getWidthHeight(this._config.orientation), listBounds + 'px'
874     );
875     this._setListContainerHeight(
876       this._config.listContainer,
877       newListItems
878     );
879     this._config.listContainer.css({
880         'position': 'relative'
881     });
882 
883     var offset = tools.getPosition(
884       this._config.orientation,
885       jQuery(this._config.listItems[this._config.activeIndex])
886     );
887     if (this._config.orientation == 'horizontal') {
888       this._config.list.css({ 'left': '-' + offset + 'px' });
889     } else {
890       this._config.list.css({ 'top': '-' + offset + 'px' });
891     }

902 /**
903  * @namespace Animation object. Moves through the list item in a linear manner.
904  *    It also slowly scrolls when the user hovers over the "previous" or
905  *    "next"-button.
906  * @namespace
907  * @param {Object} newConfig Standard configuration object.
908  * @param {Object|Null} extraConfig Optional extra configuration object.
909  * @return {Object} instance of an animation object.
910  */
911 onoPager.animation.linearScroller = function(newConfig, extraConfig) {
912   /**
913    * New animation object.
914    * @memberOf onoPager.animation.linearScroller
915    * @this
916    */
917   var linearScrollerInstance = new onoPager.animation._standard(
918     newConfig, extraConfig
919   );
920   var tools = onoPager.tools;
921   var topLeft = tools.getTopLeft(linearScrollerInstance._config.orientation);
922 
923   function checkBounds(currentMargin, list, move) {
924     var pagePrevious = linearScrollerInstance._config.pagePrevious;
925     var pageNext = linearScrollerInstance._config.pageNext;
926     var listSize = tools.getInnerSize(
927       linearScrollerInstance._config.orientation,
928       jQuery(list)
929     );
930     var viewportSize = tools.getInnerSize(
931       linearScrollerInstance._config.orientation,
932       jQuery(list).parent()
933     );
934     var maxMarginBound = listSize - viewportSize;
935     var newMargin = currentMargin - move;
936     var DISABLED = 'disabled';
937 
938     if (newMargin >= 0) {
939       pagePrevious.addClass(DISABLED);
940       return false;
941     } else if ((newMargin - 1) <= (-maxMarginBound)) {
942       pageNext.addClass(DISABLED);
943       return false;
944     } else {
945       pagePrevious.removeClass(DISABLED);
946       pageNext.removeClass(DISABLED);
947       return true;
948     }
949   }
950 
951   function currentOffset() {
952     var currentMargin = linearScrollerInstance._config.list.css(
953                           'margin-' + topLeft);
954     currentMargin = currentMargin.replace('px', '');
955     currentMargin = parseInt(currentMargin);
956     return currentMargin;
957   }
958 
959   /**
960    * @see onoPager.animation._standard#init
961    * @memberOf onoPager.animation.linearScroller
962    * @this
963    */
964   linearScrollerInstance.init = function() {
965     var listSize = 0;
966     if (this._config.orientation == 'horizontal') {
967       this._config.listItems.css('float', 'left');
968       this._config.listItems.each(function(index) {
969         listSize += jQuery(this).outerWidth(true);
970       });
971       this._config.list.css(
972         {
973           'width': listSize + 'px',
974           'position': 'relative'
975         }
976       );
977     this._config.listContainer.css('position', 'relative');
978     }
979     var viewport = tools.getInnerSize(
980       this._config.orientation,
981       this._config.list.parent()
982     );
983 
984     if (listSize > viewport) {
985       this._config.pageNext.show();
986     }
987 
988     checkBounds(currentOffset(), this._config.list, 0);
989   }

1000   /**
1001    * @see onoPager.animation._standard#pagerHover
1002    * @memberOf onoPager.animation.linearScroller
1003    * @this
1004    */
1005   linearScrollerInstance.pagerHover = function(move) {
1006     var speed = 5;
1007     if (move != 0) {
1008       var list = this._config.list;
1009       this._moveInterval = setInterval(
1010         function() {
1011           var currentMargin = currentOffset();
1012           if (!checkBounds(currentMargin, list, move)) {
1013             return;
1014           }
1015           list.css('margin-' + topLeft, (-move) + currentMargin + 'px');
1016         },
1017         speed
1018       );
1019     } else {
1020       clearInterval(this._moveInterval);
1021     }
1022   }
1023 
1024   return linearScrollerInstance;
