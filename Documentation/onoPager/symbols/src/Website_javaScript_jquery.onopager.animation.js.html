 11 /**
 12  * @namespace Animation namespace
 13  */
 14 onoPager.animation = (function() {
 15   // Throws an error if the created animation object does not comply to the
 16   // interface.
 17   function interfaceCheck(animation, publicMethods) {
 18     var hasMethod;
 19     for (var i = 0; i < publicMethods.length; i++) {
 20       hasMethod = false;
 21       for (object in animation) {
 22         if (object == publicMethods[i]) {
 23           hasMethod = true;
 24         }
 25       }
 26       if (hasMethod == false) {
 27         throw new Error('Animation object does not implement ' +
 28           'public method "' + publicMethods[i] + '"');
 29       }
 30       if (typeof(animation[object]) != 'function') {
 31         throw new Error('animation.' + publicMethods[i] + ' is not of type' +
 32           '"function", but "' + typeof(animation[object]) + '"');
 33       }
 34     }
 35   }
 36 
 37   return {
 38     /**
 39      * This method creates and returns an animation object.
 40      *
 41      * @param {String} animationType Name of the animation that must be loaded.
 42      * @param {Object} config Configuration object.
 43      * @param {Object} extraConfig Extra, custom configuration object for the
 44      *   animation object.
 45      * @return {object} The animation object.
 46      */
 47     createAnimation: function(animationType, config, extraConfig) {
 48       if (typeof(onoPager.animation[animationType]) != 'function') {
 49         throw new Error('animationType "' + animationType + '" is not of ' +
 50           'type function, but ' + typeof(onoPager.animation[animationType]));
 51       }
 52       config.root.addClass('onoPager_animation_' + animationType);
 53 
 54       var animation = onoPager.animation[animationType](
 55         {
 56           list: config.list,
 57           listContainer: config.listContainer,
 58           listContainerHeight: config.listContainerHeight,
 59           adjustHeightToListItem: config.adjustHeightToListItem,
 60           listItems: config.listItems,
 61           animationSpeed: config.animationSpeed,
 62           orientation: config.orientation,
 63           pagePerItem: config.pagePerItem,
 64           pageNext: config.pageNext,
 65           pagePrevious: config.pagePrevious,
 66           activeIndex: config.activeIndex,
 67           animationEasing: config.animationEasing,
 68           autoPage: config.autoPage
 69         },
 70         extraConfig
 71       );
 72 
 73       interfaceCheck(
 74         animation,
 75         [
 76           'page',
 77           'init',
 78           'pagerHover'
 79         ]
 80       );
 81 
 82       return animation;
 83     }
 84   };
 85 })();

 92 /**
 93  * @namespace Base component of all animation objects. All properties and
 94  * methods defined here are available to all animation objects like "linear",
 95  * "linearScroller" and "slides". All private properties and methods are
 96  * designated by an underscore prefix. For example: "_checkMaxScroll" is
 97  * designated as a private method.
 98  *
 99  * @constructor
100  * @param {Object} newConfig Standard configuration object.
101  * @param {Object} newConfig.list jQuery object. References to the UL, OL
102  *    or any element that is the parent of list items.
103  * @param {Object} newConfig.listContainer jQuery object. References to
104  *    the element that is the container for the list from which the pager is
105  *    built.
106  * @param {String} newConfig.listContainerHeight If you can check if the height
107  *    of the list container is specifically set.
108  * @param {Boolean} newConfig.adjustHeightToListItem.active. If true, the
109  *    list container will adjust itself to the height of the active list item.
110  * @param {Boolean} newConfig.adjustHeightToListItem.animate. If
111  *    true, the container will animatie to its new height. Default value is
112  *    true.
113  * @param {Object} newConfig.listItems jQuery object. Collection of all list
114  *    items. If the listContainer is an unordered list (UL), the lsitItems will
115  *    be the LI-elements.
116  * @param {Number} newConfig.animationSpeed Determines the speed at
117  *    which the animations take place.
118  * @param {String} newConfig.orientation Determines on what axis the
119  *    animation moves. Possible values are 'horizontal' and 'vertical' though
120  *    it's possible to use other values as long as the animation object
121  *    supports that value. Default value is 'horizontal'.
122  * @param {Boolean} newConfig.pagePerItem If true, each page will be as long
123  *    as one list item. If false, each page will be as long as the list
124  *    container, which may span multiple list items.
125  * @param {Object} newConfig.pageNext jQuery object. References to the page link
126  *    'next'.
127  * @param {Object} newConfig.pagePrevious jQuery object. References to the page
128  *     link 'previous'.
129  * @param {Number} newConfig.activeIndex Sets initial visible page. By
130  *    default the pager starts at index 0.
131  * @param {Object} newConfig.pager Reference to the pager instance.
132  * @param {Object} newConfig.scroller Reference to the scroller instance.
133  *    If it exists.
134  * @param {Boolean} newConfig.autoPage.active Activates auto pager.
135  * @param {Boolean} newConfig.autoPage.interval The interval between
136  *    autopaging. Time value is set in milliseconds.
137  * @param {Object|Null} extraConfig Optional extra configuration object.
138  */
139 onoPager.animation._standard = function(newConfig, extraConfig) {
140   this._config = {
141     list: null,
142     listContainer: null,
143     listContainerHeight: '',
144     adjustHeightToListItem: {},
145     listItems: null,
146     animationSpeed: 1000,
147     orientation: null,
148     pagePerItem: true,
149     pageNext: null,
150     pagePrevious: null,
151     activeIndex: 0,
152     pager: {},
153     scroller: {},
154     autoPage: {},
155     extraConfig: {}
156   };
157   var containerHeightSetCount = 0;
158 
159   jQuery.extend(true, this._config, newConfig);
160   if (typeof(extraConfig) == 'object') {
161     jQuery.extend(true, this._config.extraConfig, extraConfig);
162   }
163 
164   function setContainerHeight(_config, newIndex, isInit, parentFunction) {
165     if (_config.adjustHeightToListItem.active == false) {
166       if (isInit && _config.listContainer.height() == 0) {
167         parentFunction._setListContainerHeight(_config.listContainer,
168                                                _config.listItems);
169       }
170       return false;
171     }
172 
173     var itemHeight = 0;
174     if (typeof(newIndex) == 'number') {
175       itemHeight = jQuery(_config.listItems[newIndex]).outerHeight();
176     } else {
177       itemHeight = _config.listItems.filter('.active').outerHeight();
178     }
179 
180     if (containerHeightSetCount == 0) {
181       // Set height during init
182       if (_config.adjustHeightToListItem.active == true) {
183         if (_config.listContainerHeight == '') {
184           _config.listContainer.css(
185             {
186               height: itemHeight
187             }
188           );
189         } else {
190           _config.listContainer.css(
191             {
192               height: _config.listContainerHeight
193             }
194           );
195         }
196       }
197     } else if (_config.adjustHeightToListItem.active == true) {
198       // Set height after paging
199       if (_config.adjustHeightToListItem.animate == true) {
200         _config.listContainer.stop(true, true);
201         _config.listContainer.animate(
202           {
203             height: itemHeight
204           },
205           {
206             duration: _config.animationSpeed,
207             easing: _config.animationEasing
208           }
209         );
210       } else {
211         _config.listContainer.css(
212           {
213             height: itemHeight
214           }
215         );
216       }
217     }
218     containerHeightSetCount++;
219   }

221   /**
222    * Checks if arg_scroll is not out of bounds. Use this method if you want to
223    * make sure that the list makes use of the whole space of the viewport. The
224    * argument arg_scroll is expected to be the offset of the list, like a
225    * margin. If the offset is too high or too low, it wil return the maximum
226    * or minimum scroll value. In other instances it will just return the
227    * arg_scroll value.
228    *
229    * @param {Number} arg_scroll The offset of the list in pixels.
230    * @return {Number} Returns either arg_scroll or the maximum or minimum scroll
231    *    value.
232    * @this
233    */
234   this._checkMaxScroll = function(arg_scroll) {
235     var tools = onoPager.tools;
236     var orientation = this._config.orientation;
237     var listSize = tools.getOuterSize(orientation, this._config.list, false);
238     var listContainerSize = tools.getInnerSize(
239       orientation,
240       this._config.listContainer
241     );
242 
243     maxScroll = listSize - listContainerSize;
244     var scroll = arg_scroll;
245     if (scroll > maxScroll) {
246       scroll = maxScroll;
247     }
248     if (scroll < 0) {
249       scroll = 0;
250     }
251     return scroll;
252   }

254   /**
255    * Calculates how many times a user must page to reach the end.
256    *
257    * @return {Number} Total number of available pages. Most of the times the
258    *    return value is calculated with the private function _getPagesLength().
259    * @this
260    */
261   this.getPagesLength = function() {
262     var listContainer = this._config.listContainer;
263     var listItems = this._config.listItems;
264     var tools = onoPager.tools;
265     var orientation = this._config.orientation;
266 
267     if (this._config.pagePerItem == true) {
268       return this._config.listItems.size();
269     } else {
270       var listItemsSize = 0;
271       listItems.each(function() {
272         listItemsSize += tools.getInnerSize(orientation, jQuery(this));
273       });
274       return Math.ceil(listItemsSize /
275                        tools.getInnerSize(orientation, listContainer));
276     }
277   }

279   /**
280    * Set list container height to the list item with the most height.
281    *
282    * @param {Object} listContainer The list container.
283    * @param {Object} listItems All items in the list (typically <li>).
284    * @this
285    */
286   this._setListContainerHeight = function(listContainer, listItems) {
287     if (listItems.size() > 1 && this._config.listContainerHeight == '') {
288       var maxHeight = 0;
289       listItems.each(function() {
290         if (maxHeight < jQuery(this).outerHeight()) {
291           maxHeight = jQuery(this).outerHeight();
292         }
293       });
294       listContainer.height(maxHeight);
295     }
296   }

298   /**
299    * Add or remove the class 'active' to a list item.
300    *
301    * @param {Object|Null} index The index number of the item. If Null, the
302    *    action is applied to all list items.
303    * @param {Boolean} add true adds the class, false removes it.
304    * @this
305    */
306   this._setActiveClass = function(index, add) {
307     if (this._config.pagePerItem == true) {
308       var item;
309       if (typeof(index) == 'number') {
310         item = jQuery(this._config.listItems[index]);
311       } else {
312         item = jQuery(this._config.listItems);
313       }
314       if (add) {
315         item.addClass('active');
316       } else {
317         item.removeClass('active');
318       }
319     }
320   }

333   /**
334    * Handles drag event of the scroller handle.
335    *
336    * @param {Number} percentage The position of the handle relative to the
337    * scroller in percentage.
338    * @this
339    */
340   this.onHandleDrag = function(percentage) {
341     var list = this._config.list;
342     var listContainer = this._config.listContainer;
343     var orientation = this._config.orientation;
344     var tools = onoPager.tools;
345     var listSize = tools.getOuterSize(orientation, list, false);
346     var listContainerSize = tools.getOuterSize(orientation,
347                                                listContainer,
348                                                false);
349     var listScrollSize = listSize - listContainerSize;
350     var listScrollPosition = Math.round(-((listScrollSize / 100) * percentage));
351     var offsetKey = tools.getTopLeft(orientation);
352     var css = {};
353 
354     css['margin-' + offsetKey] = listScrollPosition + 'px';
355     list.css(css);
356   }

372   /**
373    * Makes the transition from one slide to the next. This method is not

376    *
377    * @param {Number} oldIndex Index of the current active item.

383   delete this.page;
384 
385   // _page is a wrapper method for animationinstance.page()
386   this._page = function(oldIndex, newIndex, direction) {
387     setContainerHeight(this._config, newIndex);
388     this.page(oldIndex, newIndex, direction);
389   }

396   this.onPagerCreated = function() {}
397   delete this.onPagerCreated;
398 
399   // _onPagerCreated is a wrapper method for animationinstance.onPagerCreated()
400   this._onPagerCreated = function() {
401     if (typeof(this.onPagerCreated) == 'function') {
402       this.onPagerCreated();
403     }
404   }

406   /**
407    * Handles mouseenter and mouseleave event on the "previous" and "next"-links.
408    * This method is not implemented in the base animation object, but is part of
409    * the interface. Failing to implement this method will result in an error.
410    *
411    * @param {Number} move Suggested move integer if this method will do some
412    *    animation.
413    */
414   this.pagerHover = function(move) {}
415   delete this.pagerHover;
416 
417   // _pagerHover is a wrapper method for animationinstance.pagerHover()
418   this._pagerHover = function(move) {
419     if (this._config.scroller.updateHandle) {
420       // update the handle of the scroller if one exists.
421       this._config.scroller.updateHandle();
422     }
423     this.pagerHover(move);
424   }

432 /**
433  * @namespace Animation object. Incoming and outgoing slides will always appear
434  *    next to each other. Even if they are not next to each other in the list,
435  *    like the first and last item in the list.
436  *
437  * @param {Object} newConfig Standard configuration object.
438  * @param {Object|Null} extraConfig Optional extra configuration object.
439  * @return {Object} instance of an animation object.
440  */
441 onoPager.animation.slides = function(newConfig, extraConfig) {
442   /**
443    * New animation object.
444    * @memberOf onoPager.animation.slides
445    */
446   var slidesInstance = new onoPager.animation._standard(newConfig, extraConfig);
447   var tools = onoPager.tools;
448 
449   /**
450    * @see onoPager.animation._standard#init
451    * @memberOf onoPager.animation.slides
452    * @this
453    */
454   slidesInstance.init = function() {
455     this._config.listItems.hide();
456     this._config.listItems.css(
457       {
458         'position': 'absolute'
459       }
460     );
461     jQuery(this._config.listItems[0]).show();
462   }
463 
464   /**
465    * @see onoPager.animation._standard#page
466    * @memberOf onoPager.animation.slides
467    * @this
468    */
469   slidesInstance.page = function(oldIndex, newIndex, direction) {
470     var oldItemLeft = 0;
471     var newItemLeft = 0;
472     var pageSize = tools.getOuterSize(
473       slidesInstance._config.orientation,
474       slidesInstance._config.listItems,
475       false
476     );
477 
478     if (oldIndex < newIndex) {
479       // Next
480       oldItemLeft = pageSize;
481       newItemLeft = -pageSize;
482     }
483     if (oldIndex > newIndex) {
484       // Previous
485       oldItemLeft = -pageSize;
486       newItemLeft = pageSize;
487     }
488 
489     jQuery(this._config.listItems[oldIndex]).stop(true, true);
490     jQuery(this._config.listItems[newIndex]).stop(true, true);
491 
492     this._config.listItems.each(function() {
493       var item = jQuery(this);
494       if (item.index() != oldIndex && item.index() != newIndex) {
495         item.hide();
496       }
497     });
498 
499     var newCss = {};
500     var topLeft = tools.getTopLeft(slidesInstance._config.orientation);
501     newCss['display'] = 'block';
502     newCss[topLeft] = oldItemLeft + 'px';
503     jQuery(this._config.listItems[newIndex]).css(newCss);
504 
505     var oldAni = {};
506     oldAni[topLeft] = newItemLeft + 'px';
507     jQuery(this._config.listItems[oldIndex]).animate(
508       oldAni,
509       {
510         duration: this._config.animationSpeed,
511         easing: this._config.animationEasing
512       }
513     );
514 
515     var newAni = {};
516     newAni[topLeft] = '0';
517     jQuery(this._config.listItems[newIndex])
518       .delay(this._config.animationSpeed / this._config.extraConfig.delay)
519       .animate(
520         newAni,
521         {
522           duration: this._config.animationSpeed,
523           easing: this._config.animationEasing
524         }
525       );
526   }
527 
528   /**
529    * @see onoPager.animation._standard#pagerHover
530    * @memberOf onoPager.animation.slides
531    * @this
532    */
533   slidesInstance.pagerHover = function(move) {
534     // Not implemented
535   }
536 
537   return slidesInstance;
538 };

597 /**
598  * @namespace Animation object. The old item fades away, while the new one fades
599  * in at the same time.
600  *
601  * @param {Object} newConfig Standard configuration object.
602  * @param {Object|Null} extraConfig Optional extra configuration object.
603  * @return {Object} instance of an animation object.
604  */
605 onoPager.animation.fade = function(newConfig, extraConfig) {
606   /**
607    * New animation object.
608    * @memberOf onoPager.animation.fade
609    */
610   var fadeInstance = new onoPager.animation._standard(newConfig, extraConfig);
611   var tools = onoPager.tools;
612 
613   /**
614    * @see onoPager.animation._standard#init
615    * @memberOf onoPager.animation.fade
616    * @this
617    */
618   fadeInstance.init = function() {
619     this._config.listItems.hide();
620     this._config.listItems.css(
621       {
622         position: 'absolute',
623         top: 0,
624         left: 0
625       }
626     );
627     jQuery(this._config.listItems[0]).show();
628   }
629 
630   /**
631    * @see onoPager.animation._standard#page
632    * @memberOf onoPager.animation.fade
633    * @this
634    */
635   fadeInstance.page = function(oldIndex, newIndex, direction) {
636     var oldItem = jQuery(this._config.listItems[oldIndex]);
637     var newItem = jQuery(this._config.listItems[newIndex]);
638 
639     // End all current animations.
640     oldItem.stop(true, true);
641     newItem.stop(true, true);
642 
643     // Hide all items except the two that will animate in the page transition.
644     this._config.listItems.each(function(index) {
645       if (index != oldIndex && index != newIndex) {
646         var item = jQuery(this);
647         item.hide();
648         item.css('z-index', 0);
649       }
650     });
651 
652     // Set order stack for animation.
653     oldItem.css(
654       {
655         zIndex: 1,
656         display: 'block',
657         opacity: 1
658       }
659     );
660     newItem.css(
661       {
662         zIndex: 2,
663         display: 'block',
664         opacity: 0
665       }
666     );
667 
668     oldItem.animate(
669       {
670         opacity: 0
671       },
672       {
673         duration: this._config.animationSpeed,
674         easing: this._config.animationEasing,
675         complete: function() {
676           jQuery(this).hide();
677         }
678       }
679     );
680     newItem.animate(
681       {
682         opacity: 1
683       },
684       {
685         duration: this._config.animationSpeed,
686         easing: this._config.animationEasing,
687         complete: function() {
688           jQuery(this).css('filter', '');
689         }
690       }
691     );
692   }

696    * @memberOf onoPager.animation.fade
697    * @this
698    */
699   fadeInstance.pagerHover = function(move) {
700     // Not implemented
701   }
702 
703   return fadeInstance;
704 };

711 /**
712  * @namespace Animation object. Moves through the list item in a linear manner.
713  *
714  * @param {Object} newConfig Standard configuration object.
715  * @param {Object|Null} extraConfig Optional extra configuration object.
716  * @return {Object} instance of an animation object.
717  */
718 onoPager.animation.linear = function(newConfig, extraConfig) {
719   /**
720    * New animation object.
721    * @memberOf onoPager.animation.linear
722    * @this
723    */
724   var linearInstance = new onoPager.animation._standard(newConfig, extraConfig);
725   var tools = onoPager.tools;
726 
727   /**
728    * @see onoPager.animation._standard#init
729    * @memberOf onoPager.animation.linear
730    * @this
731    */
732   linearInstance.init = function() {
733     // This animation only works with horizontal and vertical orientation.
734     if (this._config.orientation != 'horizontal' &&
735         this._config.orientation != 'vertical') {
736       throw new Error('Orientation must be either horizontal ' +
737         'or vertical. It\'s now ' + this._config.orientation);
738     }
739 
740     var listBounds = 0;
741 
742     // Set class 'active' to the visible list item if the pager is set to page
743     // per list item.
744     linearInstance._setActiveClass(linearInstance._config.activeIndex, true);
745 
746     // Set list container width
747     this._config.listItems.each(function(index) {
748       listBounds += tools.getOuterSize(
749         linearInstance._config.orientation,
750         jQuery(this)
751       );
752     });
753     this._config.listItems.css('float', 'left');
754     this._config.list.css('position', 'relative');
755     this._config.list.css(
756       tools.getWidthHeight(this._config.orientation), listBounds + 'px'
757     );
758 
759     // Set list container height
760     this._setListContainerHeight(
761       this._config.listContainer,
762       this._config.listItems
763     );

915   // Appends and prepends items until the list always fills the screen.
916   function fillIdleSpace(idleSpace) {
917     var list = linearContinuousInstance._config.list;
918     var listItems = linearContinuousInstance._config.listItems;
919 
920     fillBefore();
921     fillAfter();

923     function fillBefore() {
924       // Add items to the begin of the list
925       var prependItems = jQuery([]);
926       var prependItemsArray = [];
927       var prependSpace = 0;
928       var itemSize = 0;
929 
930       for (var i = 1; i <= listItems.size(); i++) {
931         if (prependSpace < (idleSpace * 2)) {
932           prependItemsArray.push(jQuery(listItems.get(-i)).clone(true));
933           itemSize = tools.getOuterSize(
934             linearContinuousInstance._config.orientation,
935             jQuery(listItems.get(-i)),
936             false
937           );
938           prependFill += itemSize;
939           if (i > 1) {
940             prependSpace += itemSize;
941           }
942           prependSpace += tools.getOuterSize(
943             linearContinuousInstance._config.orientation,
944             jQuery(listItems.get(-i)),
945             false
946           );
947         } else {
948           break;
949         }
950       }
951       for (var i = 0; i < prependItemsArray.length; i++) {
952         list.prepend(
953           jQuery(prependItemsArray[i]).clone(true)
954         );
955         list.find('li:first').attr('data-onopager-list-direction', -1);
956       }
957     }
958 
959     function fillAfter() {
960       // Add items at the end of the list
961       var appendItems = jQuery([]);
962       var appendSpace = 0;
963 
964       for (var i = 0; i < listItems.size(); i++) {
965         if (appendSpace < (idleSpace * 2)) {
966           appendItems = appendItems.add(
967             jQuery(listItems.get(i)).clone(true)
968                                     .attr('data-onopager-list-direction', 1)
969           );
970         } else {
971           break;
972         }
973       }
974       list.append(appendItems);
975     }

1006     if (oldIndex != newIndex) {
1007       linearContinuousInstance._setActiveClass(oldIndex, false);
1008     }
1009     this._config.list.stop(true, false);
1010 
1011     var topLeft = tools.getTopLeft(this._config.orientation);
1012     var oldItem = resetPosition(oldIndex, newIndex);
1013 
1014     var offset;
1015     if (this._config.pagePerItem == true) {

1017         this._config.orientation,
1018         jQuery(this._config.listItems[newIndex])
1019       );
1020     } else {
1021       var currentOffset = this._config.list.css(topLeft).replace('px', '');

1024       if (oldIndex == 0 && newIndex == (listSize - 1) && direction == -1) {
1025         indexMove = -1;

1027                  newIndex == 0 && direction == 1) {
1028         indexMove = 1;
1029       } else {
1030         indexMove = newIndex - oldIndex;
1031       }

1149     // To create the appearence of a list that repeats itself infinitely, the
1150     // list is repositioned just before it threatens to get out of bounds.
1151     function resetPosition(oldIndex, newIndex) {
1152       var pageOverflow = 0;
1153       var topLeft = tools.getTopLeft(
1154         linearContinuousInstance._config.orientation
1155       );
1156       var offset;
1157       var oldItem = jQuery(
1158         linearContinuousInstance._config.listItems[oldIndex]
1159       );
1160 
1161       if (oldIndex > newIndex && direction == 1) {
1162         // If user pages from last page to first page, position to the page
1163         // *before* the first page.
1164         var firstIndex = newListItems.index(
1165           linearContinuousInstance._config.listItems[newIndex]
1166         );
1167 
1168         if (linearContinuousInstance._config.pagePerItem == true) {
1169           oldItem = jQuery(newListItems[firstIndex -
1170                                         (newIndex + (listSize - oldIndex))]);
1171           offset = tools.getPosition(
1172             linearContinuousInstance._config.orientation,
1173             oldItem
1174           );
1175           offset = -Math.round(offset);
1176         } else {
1177           var maxOffset = prependFill + listItemsSize;
1178           var currentOffset = tools.getPosition(
1179             linearContinuousInstance._config.orientation,
1180             linearContinuousInstance._config.list
1181           );
1182           pageOverflow = currentOffset + maxOffset;
1183           offset = -(prependFill - containerSize +
1184                      (containerSize - pageOverflow));
1185           if (offset < -listItemsSize) {
1186             offset += listItemsSize;
1187           }
1188         }
1189       } else if (newIndex > oldIndex && direction == -1) {
1190         // If user pages from the first item to last item, position on the item
1191         // *after* the last item.
1192         var lastIndex = newListItems.index(
1193           linearContinuousInstance._config.listItems[newIndex]
1194         );
1195 
1196         if (linearContinuousInstance._config.pagePerItem == true) {
1197           oldItem = jQuery(newListItems[lastIndex +
1198                                         ((listSize - newIndex) + oldIndex)]);
1199           offset = tools.getPosition(
1200             linearContinuousInstance._config.orientation,
1201             oldItem
1202           );
1203           offset = -Math.round(offset);
1204         } else {
1205           var currentOffset = tools.getPosition(
1206             linearContinuousInstance._config.orientation,
1207             linearContinuousInstance._config.list
1208           );
1209           pageOverflow = currentOffset + prependFill;
1210           offset = -(prependFill + listItemsSize - containerSize +
1211                       (containerSize - pageOverflow));
1212           if (offset > -((rootSize * 2) - containerSize)) {
1213             offset -= listItemsSize;
1214           }
1215         }
1216       }
1217 
1218       if (offset) {
1219         linearContinuousInstance._config.list.css(topLeft, offset + 'px');
1220       }
1221 
1222       return oldItem;
1223     }

1345     var listSize = tools.getOuterSize(

1351       linearScrollerInstance._config.orientation,
1352       jQuery(list).parent()
1353     );
1354     var maxMarginBound = listSize - viewportSize;
1355     var newMargin = currentMargin - move;
1356     var DISABLED = 'onoPager_disabled';
1357 
1358     if (newMargin >= 0) {
1359       pagePrevious.addClass(DISABLED);
1360       return false;
1361     } else if ((newMargin - 1) <= (-maxMarginBound)) {
1362       pageNext.addClass(DISABLED);
1363       return false;
1364     } else {
1365       pagePrevious.removeClass(DISABLED);
1366       pageNext.removeClass(DISABLED);
1367       return true;
1368     }
