1 /**
  2  * @namespace Paging logic.
  3  *
  4  * @constructor
  5  * @param {Number} arg_index Initial index position.
  6  * @param {Number} arg_length Number of items in list.
  7  * @param {Boolean} arg_doesLoop If set to true, the pager will loop
  8  *  from the last item back to the first.
  9  * @param {Object} arg_controls Object that contains certain elements in the
 10  *  pager, wrapped in a jQuery object.
 11  * @param {Object} arg_status Configuration object for the status box.
 12  */
 13 onoPager.pager = function(arg_index,
 14                           arg_length,
 15                           arg_doesLoop,
 16                           arg_controls,
 17                           arg_status) {
 18   var index = arg_index || 0;
 19   var length = arg_length || 0;
 20   var doesLoop = (typeof(arg_doesLoop) == 'boolean') ? arg_doesLoop : true;
 21 
 22   // Set autopager variables
 23   var autoPageConfig = {};  // Autopage configuration object
 24   var autoPageInterval;     // The interval between transitions in milliseconds
 25   var autoPageAnimation;    // The auto page animation object
 26   var autoPageContainer;    // The element in which the page animation happens
 27   var listContainer;        // The element that holds the list
 28   var animationSpeed;       // The speed of the transitions in milliseconds
 29   var orientation;          // The orientation
 30 
 31   // Set pager controls
 32   var controls = {
 33     previous: null,
 34     next: null,
 35     gotoLinks: null,
 36     status: null
 37   };
 38   jQuery.extend(true, controls, arg_controls);
 39   var status = {
 40     active: null,
 41     prependText: null,
 42     seperationText: null,
 43     appendText: null
 44   };
 45   jQuery.extend(true, status, arg_status);
 46   setPagerButtons(index);
 47 
 48 
 49 
 50   // Handle index values that are out of bounds
 51   function indexCheckBounds(arg_index) {
 52     if (doesLoop == true) {
 53       return indexCheckBounds_loop(arg_index);
 54     } else {
 55       return indexCheckBounds_noLoop(arg_index);
 56     }
 57   }
 58 
 59   // Handle index values that are out of bounds in a looping pager
 60   function indexCheckBounds_loop(arg_index) {
 61     if (arg_index < 0) {
 62       return (length - 1);
 63     } else if (arg_index > (length - 1)) {
 64       return 0;
 65     } else {
 66       return arg_index;
 67     }
 68   }
 69 
 70   // Handle index values that are out of bounds pager without looping
 71   function indexCheckBounds_noLoop(arg_index) {
 72     if (arg_index < 0) {
 73       return 0;
 74     } else if (arg_index > (length - 1)) {
 75       return (length - 1);
 76     } else {
 77       return arg_index;
 78     }
 79   }
 80 
 81   // Update the paging controls
 82   function setPagerButtons(index) {
 83     var DISABLED = 'disabled';
 84     if (controls.gotoLinks) {
 85       var activeLink = controls.gotoLinks.find('a:eq(' + index + ')');
 86       activeLink.siblings().removeClass('onoPager_active');
 87       activeLink.addClass('onoPager_active');
 88     }
 89 
 90     if (controls.previous && doesLoop == false) {
 91       if (index == 0) {
 92         controls.previous.addClass(DISABLED);
 93       } else {
 94         controls.previous.removeClass(DISABLED);
 95       }
 96     }
 97     if (controls.next && doesLoop == false) {
 98       if (index == (length - 1)) {
 99         controls.next.addClass(DISABLED);
100       } else {
101         controls.next.removeClass(DISABLED);
102       }
103     }
104     if (controls.status) {
105       controls.status.html(
106         status.prependText + (index + 1) + status.seperationText +
107         length + status.appendText
108       );
109     }
110   }
111 
112   // Set active index and check if the it's out of bounds
113   function setIndex(arg_index) {
114     index = indexCheckBounds(arg_index);
115     setPagerButtons(index);
116     return index;
117   }
118 
119   // Set index by adding or substracting to the active item index
120   function move(move) {
121     index += move;
122     index = indexCheckBounds(index);
123     setPagerButtons(index);
124     return index;
125   }
126 
127   // Initializes autopage
128   function startAutopager() {
129     autoPageAnimation = setAnimation();
130     autoPageInterval = setInterval(autoPager, autoPageConfig.interval);
131   }
132 
133   // Starts a page transition (triggered by interval)
134   function autoPager() {
135     if (doesLoop == false && (index == (length - 1))) {
136       clearInterval(autoPageInterval);
137     }
138     autoPageConfig.animation._page(index, move(1));
139     autoPageAnimation._start();
140   }
141 
142   // Create autopage animation object
143   function setAnimation() {
144     if (autoPageConfig.autoPageAnimationType != '') {
145       var newAnimation = onoPager.autopageAnimation.createAnimation(
146         {
147           listContainer: listContainer,
148           animationSpeed: animationSpeed,
149           orientation: orientation,
150           root: autoPageContainer,
151           autoPageAnimationType: autoPageConfig.autoPageAnimationType,
152           autoPageInterval: autoPageConfig.interval
153         }
154       );
155       newAnimation._init();
156       return newAnimation;
157     } else {
158       return null;
159     }
160   }
161

164   /**
165    * Returns index of currently active page
166    * @return {Number} Index of currently active page.
167    */
168   this.getIndex = function() {
169     return index;
170   }
171 
172   /**
173    * Sets index by assigning it.
174    * @param {Number} arg_index The new index.
175    * @return {Number} Returns the current index.
176    */
177   this.setIndex = function(arg_index) {
178     return setIndex(arg_index);
179   }
180 
181   /**
182    * Sets index adding move value to index
183    * @param {Number} arg_move The number the pager has to move.
184    * @return {Number} index Returns the current index.
185    */
186   this.move = function(arg_move) {
187     return move(arg_move);
188   }
189 
190   /**
191    * Sets index adding move value to index
192    * @param {Object} arg_autoPageConfig AutoPage configuration.
193    * @param {Number} arg_animationSpeed The time a transition will take in
194    *    milliseconds.
195    * @param {Object} arg_animation Animation instance.
196    * @param {Object} arg_orientation Determines on what axis the
197    *    animation moves. Possible values are 'horizontal' and 'vertical' though
198    *    it's possible to use other values as long as the animation object
199    *    supports that value. Default value is 'horizontal'.
200    * @param {Object} arg_listContainer Element that holds the list.
201    * @param {Object} arg_list Element that is the root of the list. That's the
202    *  <ul> most of the time.
203    * @param {Object} arg_autoPageContainer The element in which the auto page
204    *    animation will take place.
205    * @example
206    * instance.initAutoPager(
207    *    {
208    *      active: true,
209    *      interval: 2000
210    *    }
211    * );
212    */
213   this.initAutopager = function(arg_autoPageConfig,
214                                 arg_animationSpeed,
215                                 arg_animation,
216                                 arg_orientation,
217                                 arg_listContainer,
218                                 arg_list,
219                                 arg_autoPageContainer) {
220     var tools = onoPager.tools;
221 
222     // Setting local private variables
223     animationSpeed = arg_animationSpeed;
224     listContainer = arg_listContainer;
225     orientation = arg_orientation;
226     autoPageContainer = arg_autoPageContainer;
227     jQuery.extend(true,
228                   autoPageConfig,
229                   arg_autoPageConfig,
230                   {animation: arg_animation});
231 
232     var overflow = tools.getInnerSize(orientation, listContainer) -
233                      tools.getInnerSize(orientation, arg_list);
234     if (overflow < 0) {
235       startAutopager();
236     }
237   }
238 
239   /**
240    * Sets index adding move value to index
241    * @example
242    * instance.initAutoPager(
243    *    {
244    *      active: true,
245    *      interval: 2000
246    *    }
247    * );
248    */
249   this.resetAutopager = function() {
250     clearInterval(autoPageInterval);
251     startAutopager();
252   }
253 };

260 /**
261  * @namespace Handles animation that gives a time indication of the intervals
262  *    between paging.
263  */
264 onoPager.autopageAnimation = (function() {
265   // Throws an error if the created animation object does not comply to the
266   // interface.
267   function interfaceCheck(animation, publicMethods) {
268     var hasMethod;
269     for (var i = 0; i < publicMethods.length; i++) {
270       hasMethod = false;
271       for (object in animation) {
272         if (object == publicMethods[i]) {
273           hasMethod = true;
274         }
275       }
276       if (hasMethod == false) {
277         throw new Error('Animation object does not implement ' +
278           'public method "' + publicMethods[i] + '"');
279       }
280       if (typeof(animation[object]) != 'function') {
281         throw new Error('animation.' + publicMethods[i] + ' is not of type' +
282           '"function", but "' + typeof(animation[object]) + '"');
283       }
284     }
285   }
286 
287   return {
288     /**
289      * This method creates and returns an animation object.
290      *
291      * @param {Object} config Configuration object.
292      * @return {object} The animation object.
293      */
294     createAnimation: function(config) {
295       if (typeof(onoPager.autopageAnimation[config.autoPageAnimationType]) !=
296           'function') {
297         throw new Error('autoPageAnimationType "' +
298           config.autoPageAnimationType + '" is not of type function, but ' +
299           typeof(onoPager.autopageAnimation[config.autoPageAnimationType])
300         );
301       }
302       config.root.addClass('onoPager_onoPager.autopageAnimation_' +
303         config.autoPageAnimationType);
304 
305       var animation = onoPager.autopageAnimation[
306         config.autoPageAnimationType](config);
307 
308       interfaceCheck(
309         animation,
310         [
311           'init',
312           'start'
313         ]
314       );
315 
316       return animation;
317     }
318   };
319 })();

326 /**
327  * @namespace Base component of all autopage animation objects. All properties
328  * and methods defined here are available to all animation objects. All private
329  * properties and methods are designated by an underscore prefix.
330  *
331  * @constructor
332  * @param {Object} newConfig Standard configuration object.
333  */
334 onoPager.autopageAnimation._standard = function(newConfig) {
335   this._config = {
336     listContainer: null,
337     animationSpeed: 0,
338     orientation: null,
339     root: null,
340     autoPageInterval: 0
341   };
342   jQuery.extend(true, this._config, newConfig);
343 
344   /**
345    * This method is run when the animation object is initialized. This method
346    * is not implemented in the base animation object, but is part of the
347    * interface. Failing to implement this method will result in an error.
348    */
349   this.init = function() {}
350   delete this.init;
351 
352   this._init = function() {
353     this.init();
354   }
355 
356   /**
357    * Starts the page animation. This method is not implemented in the base
358    * animation object, but is part of the interface. Failing to implement this
359    * method will result in an error.
360    */
361   this.start = function() {}
362   delete this.page;
363 
364   this._start = function() {
365     this.start();
366   }
367 };

374 /**
375  * @namespace Autopage animation object. This object will create a simple
376  * timeline
377  *
378  * @param {Object} newConfig Standard configuration object.
379  * @return {Object} instance of an animation object.
380  */
381 onoPager.autopageAnimation.timeline = function(newConfig) {
382   /**
383    * New animation object.
384    * @memberOf onoPager.autopageAnimation.timeline
385    */
386   var timelineInstance = new onoPager.autopageAnimation._standard(newConfig);
387   var tools = onoPager.tools;
388   var bar = jQuery([]);
389   var root = timelineInstance._config.root;
390   var listContainer = timelineInstance._config.listContainer;
391 
392   /**
393    * @see onoPager.autopageAnimation._standard#init
394    * @memberOf onoPager.autopageAnimation.timeline
395    * @this
396    */
397   timelineInstance.init = function() {
398     root.html('<div class="onoPager_autoPageBar"></div>');
399     bar = root.find('div.onoPager_autoPageBar');
400     root.css(
401       {
402         width: listContainer.innerWidth(),
403         position: 'absolute',
404         left: listContainer.position().left + 'px',
405         top: listContainer.position().top +
406                listContainer.innerHeight(true) + 'px'
407       }
408     );
409 
410     bar.css('width', 0);
411     bar.animate(
412       {
413         width: listContainer.innerWidth() + 'px'
414       },
415       {
416         duration: this._config.autoPageInterval,
417         easing: 'linear'
418       }
419     );
420   }
421 
422   /**
423    * @see onoPager.autopageAnimation._standard#start
424    * @memberOf onoPager.autopageAnimation.timeline
425    * @this
426    */
427   timelineInstance.start = function() {
428     var interval = this._config.autoPageInterval - this._config.animationSpeed;
429 
430     bar.stop(true, true);
431     bar.css('width', 0);
432     bar.delay(this._config.animationSpeed).animate(
433       {
434         width: listContainer.innerWidth() + 'px'
435       },
436       {
437         duration: interval,
438         easing: 'linear'
439       }
440     );
441   }
442 
443   return timelineInstance;
444 };
