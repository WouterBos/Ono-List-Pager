 12 /**
 13  * @namespace Paging logic. The pager keeps track of the index of the active
 14               list item and controls the autopager and autopager animations.
 15  *
 16  * @constructor
 17  * @param {Number} arg_index Initial index position.
 18  * @param {Number} arg_length Number of items in list.
 19  * @param {Boolean} arg_doesLoop If set to true, the pager will loop
 20  *    from the last item back to the first.
 21  * @param {Object} arg_controls Object that contains certain elements in the
 22  *    pager, wrapped in a jQuery object.
 23  * @param {Object} arg_status Configuration object for the status box.
 24  * @param {Number} arg_gotoLinks_hideThreshold Controls the number of links
 25  *    before and after the active link that are visible.
 26  */
 27 onoPager.pager = function(arg_index,
 28                           arg_length,
 29                           arg_doesLoop,
 30                           arg_controls,
 31                           arg_status,
 32                           arg_gotoLinks_hideThreshold) {
 33   var index = arg_index || 0;
 34   var length = arg_length || 0;
 35   var doesLoop = (typeof(arg_doesLoop) == 'boolean') ? arg_doesLoop : true;
 36   var gotoLinks_hideThreshold = arg_gotoLinks_hideThreshold || -1;
 37 
 38   // Set autopager variables
 39   var autoPageConfig = {};  // Autopage configuration object
 40   var autoPageInterval;     // The interval between transitions in milliseconds
 41   var autoPageAnimation;    // The auto page animation object
 42   var autoPageContainer;    // The element in which the page animation happens
 43   var listContainer;        // The element that holds the list
 44   var animationSpeed;       // The speed of the transitions in milliseconds
 45   var orientation;          // The orientation
 46   var lockDuringTransition; // Determines of a user can make a page before the
 47                             //    current transition is finished
 48   var list;                 // The pager list. Most of the time that's a UL.
 49 
 50   // Set pager controls
 51   var controls = {
 52     previous: null,
 53     next: null,
 54     gotoLinks: null,
 55     status: null
 56   };
 57   jQuery.extend(true, controls, arg_controls);
 58   var status = {
 59     active: null,
 60     prependText: null,
 61     seperationText: null,
 62     appendText: null
 63   };
 64   jQuery.extend(true, status, arg_status);
 65   setPagerButtons(index);
 66

 69   // Handle index values that are out of bounds
 70   function indexCheckBounds(arg_index) {
 71     if (doesLoop == true) {
 72       return indexCheckBounds_loop(arg_index);
 73     } else {
 74       return indexCheckBounds_noLoop(arg_index);
 75     }
 76   }
 77 
 78   // Handle index values that are out of bounds in a looping pager
 79   function indexCheckBounds_loop(arg_index) {
 80     if (arg_index < 0) {
 81       return (length - 1);
 82     } else if (arg_index > (length - 1)) {
 83       return 0;
 84     } else {
 85       return arg_index;
 86     }
 87   }
 88 
 89   // Handle index values that are out of bounds pager without looping
 90   function indexCheckBounds_noLoop(arg_index) {
 91     if (arg_index < 0) {
 92       return 0;
 93     } else if (arg_index > (length - 1)) {
 94       return (length - 1);
 95     } else {
 96       return arg_index;
 97     }
 98   }
 99 
100   // Update the paging controls
101   function setPagerButtons(index) {
102     var DISABLED = 'onoPager_disabled';
103     var ENABLED = 'onoPager_enabled';
104     if (controls.gotoLinks) {
105       var activeLink = controls.gotoLinks.find('a:eq(' + index + ')');
106       activeLink.siblings().removeClass('onoPager_active');
107       activeLink.addClass('onoPager_active');
108       if (gotoLinks_hideThreshold >= 0) {
109         toggleGotoLinks();
110       }
111     }
112 
113     if (controls.previous && doesLoop == false) {
114       if (index == 0) {
115         controls.previous.addClass(DISABLED).removeClass(ENABLED);
116       } else {
117         controls.previous.removeClass(DISABLED).addClass(ENABLED);
118       }
119     }
120     if (controls.next && doesLoop == false) {
121       if (index == (length - 1)) {
122         controls.next.addClass(DISABLED).removeClass(ENABLED);
123       } else {
124         controls.next.removeClass(DISABLED).addClass(ENABLED);
125       }
126     }
127     if (controls.status) {
128       controls.status.html(
129         status.prependText + (index + 1) + status.seperationText +
130         length + status.appendText
131       );
132     }
133 
134     function toggleGotoLinks() {
135       controls.gotoLinks.find('a').hide();
136       //activeLink.show();
137 
138       var firstVisibleItem = index - gotoLinks_hideThreshold;
139       var totalVisibleItems = (gotoLinks_hideThreshold * 2) + 1;
140       if (firstVisibleItem < 0) {
141         firstVisibleItem = 0;
142       }
143       if (firstVisibleItem > (length - totalVisibleItems)) {
144         firstVisibleItem = (length - totalVisibleItems);
145       }
146 
147       for (var i = firstVisibleItem;
148            i < firstVisibleItem + totalVisibleItems;
149            i++) {
150         controls.gotoLinks.find('a:eq(' + i + ')').show();
151       }
152     }
153   }
154 
155   // Set active index and check if the it's out of bounds
156   function setIndex(arg_index) {
157     index = indexCheckBounds(arg_index);
158     setPagerButtons(index);
159     return index;
160   }
161 
162   // Set index by adding or substracting to the active item index
163   function move(move) {
164     index += move;
165     index = indexCheckBounds(index);
166     setPagerButtons(index);
167     return index;
168   }

170   // Initializes autopage
171   function startAutopager() {
172     if (!autoPageAnimation) {
173       autoPageAnimation = setAnimation();
174     } else {
175       autoPageAnimation.start();
176     }
177     autoPageInterval = setInterval(autoPager, autoPageConfig.interval);
178   }
179 
180   // Starts a page transition (triggered by interval)
181   function autoPager() {
182     // If the root element of the onoPager is removed in the DOM, the interval
183     // has to be removed.
184     if (listContainer.closest('div.onoPager').size() == 0) {
185       clearInterval(autoPageInterval);
186     }
187 
188     var canPage = onoPager.tools.canPage(
189       listContainer.closest('div.onoPager'),
190       lockDuringTransition,
191       list,
192       list.find('> *.onoPager_listItem')
193     );
194     var canAutoPage = onoPager.tools.canAutoPage(
195       listContainer.closest('div.onoPager')
196     );
197 
198     if (canPage == true && canAutoPage == true) {
199       if (doesLoop == false && (index == (length - 1))) {
200         clearInterval(autoPageInterval);
201       }
202       autoPageConfig.animation._page(index, move(1), 1);
203       if (autoPageAnimation) {
204         autoPageAnimation._start();
205       }
206     }
207   }
208 
209   // Create autopage animation object
210   function setAnimation() {
211     if (autoPageConfig.autoPageAnimationType != '') {
212       var newAnimation = onoPager.autopageAnimation.createAnimation(
213         {
214           listContainer: listContainer,
215           animationSpeed: animationSpeed,
216           orientation: orientation,
217           root: autoPageContainer,
218           autoPageAnimationType: autoPageConfig.autoPageAnimationType,
219           autoPageInterval: autoPageConfig.interval
220         },
221         autoPageConfig.extraConfig
222       );
223       // In some cases onoPager.autopageAnimation.createAnimation returns
224       // an undefined object because the animation object is not compatible
225       // with the browser it runs in.
226       if (newAnimation) {
227         newAnimation._init();
228         return newAnimation;
229       } else {
230         autoPageConfig.autoPageAnimationType = '';
231         return null;
232       }
233     } else {
234       return null;
235     }
236   }

266   /**
267    * Sets index adding move value to index
268    * @param {Object} arg_autoPageConfig AutoPage configuration.
269    * @param {Number} arg_animationSpeed The time a transition will take in
270    *    milliseconds.
271    * @param {Object} arg_animation Animation instance.
272    * @param {Object} arg_orientation Determines on what axis the
273    *    animation moves. Possible values are 'horizontal' and 'vertical' though
274    *    it's possible to use other values as long as the animation object
275    *    supports that value. Default value is 'horizontal'.
276    * @param {Object} arg_listContainer Element that holds the list.
277    * @param {Object} arg_list Element that is the root of the list. That's the
278    *  <ul> most of the time.
279    * @param {Object} arg_autoPageContainer The element in which the auto page
280    *    animation will take place.
281    * @param {Boolean} arg_lockDuringTransition Determines of a user can make a
282    *    page before the current transition is finished.
283    * @example
284    * instance.initAutoPager(
285    *    {
286    *      active: true,
287    *      interval: 2000
288    *    }
289    * );
290    */
291   this.initAutopager = function(arg_autoPageConfig,
292                                 arg_animationSpeed,
293                                 arg_animation,
294                                 arg_orientation,
295                                 arg_listContainer,
296                                 arg_list,
297                                 arg_autoPageContainer,
298                                 arg_lockDuringTransition) {
299     var tools = onoPager.tools;
300 
301     // Setting local private variables
302     animationSpeed = arg_animationSpeed;
303     listContainer = arg_listContainer;
304     orientation = arg_orientation;
305     autoPageContainer = arg_autoPageContainer;
306     lockDuringTransition = arg_lockDuringTransition;
307     list = arg_list;
308     jQuery.extend(true,
309                   autoPageConfig,
310                   arg_autoPageConfig,
311                   {animation: arg_animation});
312     var listSize = 0;
313     list.find('*.onoPager_listItem').each(function() {
314       listSize += tools.getOuterSize(orientation, jQuery(this), false);
315     });
316     var overflow = tools.getOuterSize(orientation,
317                                       listContainer,
318                                       false) - listSize;
319     if (overflow < 0) {
320       startAutopager();
321     }
322 
323     if (autoPageConfig.pauseOnHover == true) {
324       arg_listContainer.closest('.onoPager').mouseenter(function() {
325         jQuery(this).addClass('onoPager_autoPageDisabled');
326       });
327       arg_listContainer.closest('.onoPager').mouseleave(function() {
328         jQuery(this).removeClass('onoPager_autoPageDisabled');
329       });
330     }
331   }
332 
333   /**
334    * Sets index adding move value to index
335    * @example
336    * instance.initAutoPager(
337    *    {
338    *      active: true,
339    *      interval: 2000
340    *    }
341    * );
342    */
343   this.resetAutopager = function() {
344     clearInterval(autoPageInterval);
345     startAutopager();
346   }
