1 /**
  2  * @namespace Paging logic.
  3  *
  4  * @constructor
  5  * @param {Number} arg_index Initial index position.
  6  * @param {Number} arg_length Number of items in list.
  7  * @param {Boolean} arg_doesLoop If set to true, the pager will loop
  8  *    from the last item back to the first.
  9  * @param {Object} arg_controls Object that contains certain elements in the
 10  *    pager, wrapped in a jQuery object.
 11  * @param {Object} arg_status Configuration object for the status box.
 12  * @param {Number} arg_gotoLinks_hideThreshold Controls the number of links
 13  *    before and after the active link that are visible.
 14  */
 15 onoPager.pager = function(arg_index,
 16                           arg_length,
 17                           arg_doesLoop,
 18                           arg_controls,
 19                           arg_status,
 20                           arg_gotoLinks_hideThreshold) {
 21   var index = arg_index || 0;
 22   var length = arg_length || 0;
 23   var doesLoop = (typeof(arg_doesLoop) == 'boolean') ? arg_doesLoop : true;
 24   var gotoLinks_hideThreshold = arg_gotoLinks_hideThreshold || -1;
 25 
 26   // Set autopager variables
 27   var autoPageConfig = {};  // Autopage configuration object
 28   var autoPageInterval;     // The interval between transitions in milliseconds
 29   var autoPageAnimation;    // The auto page animation object
 30   var autoPageContainer;    // The element in which the page animation happens
 31   var listContainer;        // The element that holds the list
 32   var animationSpeed;       // The speed of the transitions in milliseconds
 33   var orientation;          // The orientation
 34   var lockDuringTransition; // Determines of a user can make a page before the
 35                             //    current transition is finished
 36   var list;                 // The pager list. Most of the time that's a UL.
 37 
 38   // Set pager controls
 39   var controls = {
 40     previous: null,
 41     next: null,
 42     gotoLinks: null,
 43     status: null
 44   };
 45   jQuery.extend(true, controls, arg_controls);
 46   var status = {
 47     active: null,
 48     prependText: null,
 49     seperationText: null,
 50     appendText: null
 51   };
 52   jQuery.extend(true, status, arg_status);
 53   setPagerButtons(index);
 54 
 55 
 56 
 57   // Handle index values that are out of bounds
 58   function indexCheckBounds(arg_index) {
 59     if (doesLoop == true) {
 60       return indexCheckBounds_loop(arg_index);
 61     } else {
 62       return indexCheckBounds_noLoop(arg_index);
 63     }
 64   }
 65 
 66   // Handle index values that are out of bounds in a looping pager
 67   function indexCheckBounds_loop(arg_index) {
 68     if (arg_index < 0) {
 69       return (length - 1);
 70     } else if (arg_index > (length - 1)) {
 71       return 0;
 72     } else {
 73       return arg_index;
 74     }
 75   }
 76 
 77   // Handle index values that are out of bounds pager without looping
 78   function indexCheckBounds_noLoop(arg_index) {
 79     if (arg_index < 0) {
 80       return 0;
 81     } else if (arg_index > (length - 1)) {
 82       return (length - 1);
 83     } else {
 84       return arg_index;
 85     }
 86   }
 87 
 88   // Update the paging controls
 89   function setPagerButtons(index) {
 90     var DISABLED = 'disabled';
 91     if (controls.gotoLinks) {
 92       var activeLink = controls.gotoLinks.find('a:eq(' + index + ')');
 93       activeLink.siblings().removeClass('onoPager_active');
 94       activeLink.addClass('onoPager_active');
 95       if (gotoLinks_hideThreshold >= 0) {
 96         toggleGotoLinks();
 97       }
 98     }
 99 
100     if (controls.previous && doesLoop == false) {
101       if (index == 0) {
102         controls.previous.addClass(DISABLED);
103       } else {
104         controls.previous.removeClass(DISABLED);
105       }
106     }
107     if (controls.next && doesLoop == false) {
108       if (index == (length - 1)) {
109         controls.next.addClass(DISABLED);
110       } else {
111         controls.next.removeClass(DISABLED);
112       }
113     }
114     if (controls.status) {
115       controls.status.html(
116         status.prependText + (index + 1) + status.seperationText +
117         length + status.appendText
118       );
119     }
120 
121     function toggleGotoLinks() {
122       controls.gotoLinks.find('a').hide();
123       //activeLink.show();
124 
125       var firstVisibleItem = index - gotoLinks_hideThreshold;
126       var totalVisibleItems = (gotoLinks_hideThreshold * 2) + 1;
127       if (firstVisibleItem < 0) {
128         firstVisibleItem = 0;
129       }
130       if (firstVisibleItem > (length - totalVisibleItems)) {
131         firstVisibleItem = (length - totalVisibleItems);
132       }
133 
134       for (var i = firstVisibleItem;
135            i < firstVisibleItem + totalVisibleItems;
136            i++) {
137         controls.gotoLinks.find('a:eq(' + i + ')').show();
138       }
139     }
140   }
141 
142   // Set active index and check if the it's out of bounds
143   function setIndex(arg_index) {
144     index = indexCheckBounds(arg_index);
145     setPagerButtons(index);
146     return index;
147   }
148 
149   // Set index by adding or substracting to the active item index
150   function move(move) {
151     index += move;
152     index = indexCheckBounds(index);
153     setPagerButtons(index);
154     return index;
155   }
156 
157   // Initializes autopage
158   function startAutopager() {
159     autoPageAnimation = setAnimation();
160     autoPageInterval = setInterval(autoPager, autoPageConfig.interval);
161   }
162 
163   // Starts a page transition (triggered by interval)
164   function autoPager() {
165     var canPage = onoPager.tools.canPage(
166       autoPageContainer.closest('div.onoPager'),
167       lockDuringTransition,
168       list,
169       list.find('> *.onoPager_listItem')
170     );
171 
172     if (canPage) {
173       if (doesLoop == false && (index == (length - 1))) {
174         clearInterval(autoPageInterval);
175       }
176       autoPageConfig.animation._page(index, move(1), 1);
177       if (autoPageAnimation) {
178         autoPageAnimation._start();
179       }
180     }
181   }
182 
183   // Create autopage animation object
184   function setAnimation() {
185     if (autoPageConfig.autoPageAnimationType != '') {
186       var newAnimation = onoPager.autopageAnimation.createAnimation(
187         {
188           listContainer: listContainer,
189           animationSpeed: animationSpeed,
190           orientation: orientation,
191           root: autoPageContainer,
192           autoPageAnimationType: autoPageConfig.autoPageAnimationType,
193           autoPageInterval: autoPageConfig.interval
194         },
195         autoPageConfig.extraConfig
196       );
197       newAnimation._init();
198       return newAnimation;
199     } else {
200       return null;
201     }
202   }

232   /**
233    * Sets index adding move value to index
234    * @param {Object} arg_autoPageConfig AutoPage configuration.
235    * @param {Number} arg_animationSpeed The time a transition will take in
236    *    milliseconds.
237    * @param {Object} arg_animation Animation instance.
238    * @param {Object} arg_orientation Determines on what axis the
239    *    animation moves. Possible values are 'horizontal' and 'vertical' though
240    *    it's possible to use other values as long as the animation object
241    *    supports that value. Default value is 'horizontal'.
242    * @param {Object} arg_listContainer Element that holds the list.
243    * @param {Object} arg_list Element that is the root of the list. That's the
244    *  <ul> most of the time.
245    * @param {Object} arg_autoPageContainer The element in which the auto page
246    *    animation will take place.
247    * @param {Boolean} arg_lockDuringTransition Determines of a user can make a
248    *    page before the current transition is finished.
249    * @example
250    * instance.initAutoPager(
251    *    {
252    *      active: true,
253    *      interval: 2000
254    *    }
255    * );
256    */
257   this.initAutopager = function(arg_autoPageConfig,
258                                 arg_animationSpeed,
259                                 arg_animation,
260                                 arg_orientation,
261                                 arg_listContainer,
262                                 arg_list,
263                                 arg_autoPageContainer,
264                                 arg_lockDuringTransition) {
265     var tools = onoPager.tools;
266 
267     // Setting local private variables
268     animationSpeed = arg_animationSpeed;
269     listContainer = arg_listContainer;
270     orientation = arg_orientation;
271     autoPageContainer = arg_autoPageContainer;
272     lockDuringTransition = arg_lockDuringTransition;
273     list = arg_list;
274     jQuery.extend(true,
275                   autoPageConfig,
276                   arg_autoPageConfig,
277                   {animation: arg_animation});
278     var listSize = 0;
279     list.find('*.onoPager_listItem').each(function() {
280       listSize += tools.getInnerSize(orientation, jQuery(this));
281     });
282     var overflow = tools.getInnerSize(orientation, listContainer) - listSize;
283     if (overflow < 0) {
284       startAutopager();
285     }
286   }
287 
288   /**
289    * Sets index adding move value to index
290    * @example
291    * instance.initAutoPager(
292    *    {
293    *      active: true,
294    *      interval: 2000
295    *    }
296    * );
297    */
298   this.resetAutopager = function() {
299     clearInterval(autoPageInterval);
300     startAutopager();
301   }
302 };

309 /**
310  * @namespace Handles animation that gives a time indication of the intervals
311  *    between paging.
312  */
313 onoPager.autopageAnimation = (function() {
314   // Throws an error if the created animation object does not comply to the
315   // interface.
316   function interfaceCheck(animation, publicMethods) {
317     var hasMethod;
318     for (var i = 0; i < publicMethods.length; i++) {
319       hasMethod = false;
320       for (object in animation) {
321         if (object == publicMethods[i]) {
322           hasMethod = true;
323         }
324       }
325       if (hasMethod == false) {
326         throw new Error('Animation object does not implement ' +
327           'public method "' + publicMethods[i] + '"');
328       }
329       if (typeof(animation[object]) != 'function') {
330         throw new Error('animation.' + publicMethods[i] + ' is not of type' +
331           '"function", but "' + typeof(animation[object]) + '"');
332       }
333     }
334   }
335 
336   return {
337     /**
338      * This method creates and returns an animation object.
339      *
340      * @param {Object} config Configuration object.
341      * @return {object} The animation object.
342      */
343     createAnimation: function(config, extraConfig) {
344       if (typeof(onoPager.autopageAnimation[config.autoPageAnimationType]) !=
345           'function') {
346         throw new Error('autoPageAnimationType "' +
347           config.autoPageAnimationType + '" is not of type function, but ' +
348           typeof(onoPager.autopageAnimation[config.autoPageAnimationType])
349         );
350       }
351       config.root.addClass('onoPager_autopageAnimation_' +
352         config.autoPageAnimationType);
353 
354       var animation = onoPager.autopageAnimation[
355         config.autoPageAnimationType](config, extraConfig);
356 
357       interfaceCheck(
358         animation,
359         [
360           'init',
361           'start'
362         ]
363       );
364 
365       return animation;
366     }
367   };
368 })();

376  * @namespace Base component of all autopage animation objects. All properties
377  * and methods defined here are available to all animation objects. All private
378  * properties and methods are designated by an underscore prefix.
379  *
380  * @constructor
381  * @param {Object} newConfig Standard configuration object.
382  */
383 onoPager.autopageAnimation._standard = function(newConfig) {
384   this._config = {
385     listContainer: null,
386     animationSpeed: 0,
387     orientation: null,
388     root: null,
389     autoPageInterval: 0
390   };
391   jQuery.extend(true, this._config, newConfig);
392 
393   /**
394    * This method is run when the animation object is initialized. This method
395    * is not implemented in the base animation object, but is part of the
396    * interface. Failing to implement this method will result in an error.
397    */
398   this.init = function() {}
399   delete this.init;
400 
401   this._init = function() {
402     this.init();
403   }

411   delete this.page;
412 
413   this._start = function() {
414     this.start();
415   }

423 /**
424  * @namespace Autopage animation object. This object will create a simple
425  * timeline
426  *
427  * @param {Object} newConfig Standard configuration object.
428  * @return {Object} instance of an animation object.
429  */
430 onoPager.autopageAnimation.timeline = function(newConfig) {
431   /**
432    * New animation object.
433    * @memberOf onoPager.autopageAnimation.timeline
434    */
435   var timelineInstance = new onoPager.autopageAnimation._standard(newConfig);
436   var tools = onoPager.tools;
437   var bar = jQuery([]);
438   var root = timelineInstance._config.root;
439   var listContainer = timelineInstance._config.listContainer;
440 
441   /**
442    * @see onoPager.autopageAnimation._standard#init
443    * @memberOf onoPager.autopageAnimation.timeline
444    * @this
445    */
446   timelineInstance.init = function() {
447     root.html('<div class="onoPager_autoPageBar"></div>');
448     bar = root.find('div.onoPager_autoPageBar');
449     root.css(
450       {
451         width: listContainer.innerWidth(),
452         position: 'absolute',
453         /* left: listContainer.position().left + 'px', */
454         top: listContainer.position().top +
455                listContainer.innerHeight(true) + 'px'
456       }
457     );
458 
459     bar.css('width', 0);
460     bar.animate(
461       {
462         width: root.innerWidth() + 'px'
463       },
464       {
465         duration: this._config.autoPageInterval,
466         easing: 'linear'
467       }
468     );
469   }
470 
471   /**
472    * @see onoPager.autopageAnimation._standard#start
473    * @memberOf onoPager.autopageAnimation.timeline
474    * @this
475    */
476   timelineInstance.start = function() {
477     var interval = this._config.autoPageInterval - this._config.animationSpeed;
478 
479     bar.stop(true, true);
480     bar.css('width', 0);
481     bar.delay(this._config.animationSpeed).animate(
482       {
483         width: root.innerWidth() + 'px'
484       },
485       {
486         duration: interval,
487         easing: 'linear'
488       }
489     );
490   }
491 
492   return timelineInstance;
493 };

500 /**
501  * @namespace Autopage animation object. This object will create a pie chart
502  * representing a clock.
503  *
504  * @param {Object} newConfig Standard configuration object.
505  * @param {Object} arg_extraConfig Extra configuration options that are
506  *    specific to this animation object.
507  * @param {Number} arg_extraConfig.widthHeight The width and height of the
508  *    clock.
509  * @param {String} arg_extraConfig.color The hex color of the clock.
510  * @param {Number} arg_extraConfig.shadowBlur The blur range of the shadow in
511  *    pixels. Default value is 5.
512  * @param {Number} arg_extraConfig.shadowOffsetX The X-axis offset of the
513  *    shadow. A positive value will cast a shadow to the right, a negative to
514  *    the left. Default value is 2.
515  * @param {Number} arg_extraConfig.shadowOffsetY The Y-axis offset of the
516  *    shadow. A positive value will cast a shadow above the clock, a negative
517  *    one below. Default value is 2.
518  * @param {String} arg_extraConfig.shadowBackgroundColor The hex color of the
519  *    shadow.
520  * @param {Number} arg_extraConfig.intervalPrecision Determines the number of
521  *    frames in the animation. A value of 1 will give the maximum smoothness
522  *    to the animation, but the clock might then not be able to keep up with
523  *    the pager's speed. A higher number will create a more responsive
524  *    animation, but also creates less frames.. The default value is 4.
525  * @return {Object} instance of an animation object.
526  */
527 onoPager.autopageAnimation.clock = function(newConfig, arg_extraConfig) {
528   /**
529    * New animation object.
530    * @memberOf onoPager.autopageAnimation.clock
531    */
532   var clockInstance = new onoPager.autopageAnimation._standard(newConfig);
533   var extraConfig = {
534     widthHeight: 16,
535     color: '#ffffff',
536     shadowBlur: 5,
537     shadowOffsetX: 2,
538     shadowOffsetY: 2,
539     shadowBackgroundColor: '#999999',
540     intervalPrecision: 4
541   };
542   jQuery.extend(true, extraConfig, arg_extraConfig);
543   var canvasWidth = extraConfig.widthHeight + extraConfig.shadowBlur;
544   if (extraConfig.shadowOffsetX < 0) {
545     canvasWidth += -extraConfig.shadowOffsetX;
546   } else {
547     canvasWidth += extraConfig.shadowOffsetX;
548   }
549 
550   var canvasHeight = extraConfig.widthHeight + extraConfig.shadowBlur;
551   if (extraConfig.shadowOffsetY < 0) {
552     canvasHeight += -extraConfig.shadowOffsetY;
553   } else {
554     canvasHeight += extraConfig.shadowOffsetY;
555   }
556 
557   var degrees;
558   var root = clockInstance._config.root;
559   var canvas;
560   var context;
561   var drawClockInterval;
562   var drawClockTimeout;
563   var listContainer = clockInstance._config.listContainer;
564   var interval;
565 
566   function drawClock() {
567       degrees += extraConfig.intervalPrecision;
568 
569       var centerX = Math.floor((extraConfig.widthHeight / 2) +
570                                extraConfig.shadowOffsetX);
571       var centerY = Math.floor((extraConfig.widthHeight / 2) +
572                                extraConfig.shadowOffsetY);
573       var radius = Math.floor(extraConfig.widthHeight / 2);
574 
575       context.clearRect(0, 0, canvasWidth + 10, canvasHeight + 10);
576       context.beginPath();
577       context.moveTo(centerX, centerY);
578       context.arc(centerX,
579                   centerY,
580                   radius,
581                   (Math.PI / 180) * -90,
582                   (Math.PI / 180) * degrees,
583                   false);
584       context.lineTo(centerX, centerY);
585       context.closePath();
586 
587       context.fillStyle = extraConfig.color;
588       context.shadowColor = extraConfig.shadowBackgroundColor;
589       context.shadowBlur = extraConfig.shadowBlur;
590       context.shadowOffsetX = extraConfig.shadowOffsetX;
591       context.shadowOffsetY = extraConfig.shadowOffsetY;
592       context.fill();
593   };
594 
595   /**
596    * @see onoPager.autopageAnimation._standard#init
597    * @memberOf onoPager.autopageAnimation.timeline
598    * @this
599    */
600   clockInstance.init = function() {
601     interval = this._config.autoPageInterval - this._config.animationSpeed;
602     root.html('<canvas width="' + canvasWidth + '" height="' +
603               canvasHeight + '"></canvas>');
604     canvas = root.find('canvas')[0];
605     context = canvas.getContext('2d');
606     this.start();
607   }
608 
609   /**
610    * @see onoPager.autopageAnimation._standard#start
611    * @memberOf onoPager.autopageAnimation.timeline
612    * @this
613    */
614   clockInstance.start = function() {
615     clearInterval(drawClockInterval);
616     degrees = -80;
617     context.clearRect(0, 0, canvasWidth, canvasHeight);
618     drawClockTimeout = setTimeout(
619       function() {
620         drawClockInterval = setInterval(
621           drawClock,
622           Math.floor(interval / (360 / extraConfig.intervalPrecision))
623         );
624       },
625       this._config.animationSpeed);
626   }
627 
628   return clockInstance;
629 };
