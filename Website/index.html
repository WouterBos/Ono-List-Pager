  listContainer: {

  },
  listItems: {
      width: '230px',
      height: '80px'
  },
  keyTriggersPage: true,
  swipeTriggersPage: true,
  animationType: 'linear',
  animationEasing: 'easeOutCubic',
  orientation: 'horizontal',
  animationSpeed: 1000
});

  • List 2: "linearScroller"

    code 
    jQuery('#list2').onoPager({
  pagePerItem: true,
  orientation: 'vertical',
  listContainer: {
    width: '250px',
    height: '100px'
  },
  listItems: {
      width: '230px',
      height: '80px'
  },
  status: {
    active: false
  },
  labels: {
    next: '»',
    previous: '«'
  },
  scroller: {
    active: true,
    pixelMove: 2
  },
  pageByNumber: false,
  animationType: 'linearScroller',
  animationSpeed: 1000
});
    • With the linearScroller the user can either scroll through the list by hovering over the 'next' or 'previous' button or dragging the handler of the scroller.
